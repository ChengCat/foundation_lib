74 static FORCEINLINE int64_t atomic_load64( atomic64_t* val )
75 {
76 #if FOUNDATION_ARCH_X86
77  uint64_t result;
78 # if FOUNDATION_COMPILER_MSVC || FOUNDATION_COMPILER_INTEL
79  __asm
80  {
81  mov esi, val;
82  mov ebx, eax;
83  mov ecx, edx;
84  lock cmpxchg8b [esi];
85  mov dword ptr result, eax;
86  mov dword ptr result[4], edx;
87  }
88 # else
89  __asm volatile(
90  "movl %%ebx, %%eax\n"
91  "movl %%ecx, %%edx\n"
92  "lock; cmpxchg8b %1"
93  : "=&A"(result)
94  : "m"(val->nonatomic));
95 # endif
96  return result;
97 #else
98  return val->nonatomic;
99 #endif
100 }

116 {
117 #if FOUNDATION_ARCH_X86
118 # if FOUNDATION_COMPILER_MSVC || FOUNDATION_COMPILER_INTEL

122  mov ebx, dword ptr val;

124  retry:

183 {

186  long long ref;
