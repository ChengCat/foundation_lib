206 atomic_load64(atomic64_t* val) {
207 #if FOUNDATION_ARCH_X86
208  int64_t result;
209 # if FOUNDATION_COMPILER_MSVC || FOUNDATION_COMPILER_INTEL
210  __asm {
211  mov esi, val;
212  mov ebx, eax;
213  mov ecx, edx;
214  lock cmpxchg8b [esi];
215  mov dword ptr result, eax;
216  mov dword ptr result[4], edx;
217  }
218 # else
219  __asm volatile(
220  "movl %%ebx, %%eax\n"
221  "movl %%ecx, %%edx\n"
222  "lock; cmpxchg8b %1"
223  : "=&A"(result)

225 # endif
