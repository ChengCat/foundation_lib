164  IP_ADAPTER_INFO adapter_info[16];
165  unsigned int status, i, j;
166  unsigned long buffer_length;
167  DWORD (STDCALL *fn_get_adapters_info)( PIP_ADAPTER_INFO, PULONG ) = 0;
168 
169  if( !_system_library_iphlpapi )
170  _system_library_iphlpapi = library_load( "iphlpapi" );
171  if( _system_library_iphlpapi )
172  fn_get_adapters_info = (DWORD (STDCALL *)( PIP_ADAPTER_INFO, PULONG ))library_symbol( _system_library_iphlpapi, "GetAdaptersInfo" );
173  if( !fn_get_adapters_info )
174  return 0;
175 
176  buffer_length = sizeof( adapter_info ); // Save memory size of buffer
177  memset( adapter_info, 0, sizeof( adapter_info ) );
178  status = fn_get_adapters_info( adapter_info, &buffer_length );
179  if( status == ERROR_SUCCESS ) for( i = 0; i < 16; ++i )

502 uint32_t system_locale( void )
503 {
504  uint32_t localeval = 0;
505  char localestr[4];
506 
507  const char* locale = config_string( HASH_USER, HASH_LOCALE );
508  if( ( locale == LOCALE_BLANK ) || ( string_length( locale ) != 4 ) )
509  locale = config_string( HASH_APPLICATION, HASH_LOCALE );
510  if( ( locale == LOCALE_BLANK ) || ( string_length( locale ) != 4 ) )
511  locale = config_string( HASH_FOUNDATION, HASH_LOCALE );
512  if( ( locale == LOCALE_BLANK ) || ( string_length( locale ) != 4 ) )
513  return _system_user_locale();
514 
515 #define _LOCALE_CHAR_TO_LOWERCASE(x) (((unsigned char)(x) >= 'A') && ((unsigned char)(x) <= 'Z')) ? (((unsigned char)(x)) | (32)) : (x)
516 #define _LOCALE_CHAR_TO_UPPERCASE(x) (((unsigned char)(x) >= 'a') && ((unsigned char)(x) <= 'z')) ? (((unsigned char)(x)) & (~32)) : (x)
517  localestr[0] = _LOCALE_CHAR_TO_LOWERCASE( locale[0] );
518  localestr[1] = _LOCALE_CHAR_TO_LOWERCASE( locale[1] );
519  localestr[2] = _LOCALE_CHAR_TO_UPPERCASE( locale[2] );
520  localestr[3] = _LOCALE_CHAR_TO_UPPERCASE( locale[3] );
521 
522  memcpy( &localeval, localestr, 4 );
523  return localeval;
524 }
