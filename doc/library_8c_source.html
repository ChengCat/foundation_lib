71 object_t library_load( const char* name )
72 {
73  library_t* library;
74  hash_t namehash;
75  unsigned int i, size;
76  uint64_t id;
77 #if FOUNDATION_PLATFORM_WINDOWS
78  char* dllname;
79  HANDLE dll;
80 #endif
81 
82  //Locate already loaded library
83  library = 0;
84  namehash = string_hash( name );
85  for( i = 0, size = objectmap_size( _library_map ); i < size; ++i )
86  {
87  library = objectmap_raw_lookup( _library_map, i );
88  if( library && ( library->namehash == namehash ) )
89  {
90  FOUNDATION_ASSERT( string_equal( library->name, name ) );
91  atomic_incr32( &library->ref );
92  return library->id;
93  }
94  }
95 
96  error_context_push( "loading library", name );
97 
98  //Try loading library
99 #if FOUNDATION_PLATFORM_WINDOWS
100 
101  dllname = string_format( "%s.dll", name );
102  dll = LoadLibraryA( dllname );
103  if( !dll )
104  {
105 #if FOUNDATION_ARCH_X86
106  string_deallocate( dllname );
107  dllname = string_format( "%s32.dll", name );
108  dll = LoadLibraryA( dllname );
109 #elif FOUNDATION_ARCH_X86_64
110  string_deallocate( dllname );
111  dllname = string_format( "%s64.dll", name );
112  dll = LoadLibraryA( dllname );
113 #endif
114  }
115  string_deallocate( dllname );
116  if( !dll )
117  {
118  log_warnf( 0, WARNING_SUSPICIOUS, "Unable to load DLL '%s': %s", name, system_error_message( 0 ) );
119  error_context_pop();
120  return 0;
121  }
122 
123 #elif FOUNDATION_PLATFORM_POSIX
124 
125 # if FOUNDATION_PLATFORM_APPLE
126  char* libname = string_format( "lib%s.dylib", name );
127 # else
128  char* libname = string_format( "lib%s.so", name );
129 # endif
130  void* lib = dlopen( libname, RTLD_LAZY );
131  string_deallocate( libname );
132 #if FOUNDATION_PLATFORM_ANDROID
133  if( !lib )
134  {
135  libname = string_format( "%s/lib%s.so", environment_executable_directory(), name );
136  lib = dlopen( libname, RTLD_LAZY );
137  string_deallocate( libname );
138  }
139 #endif
140  if( !lib )
141  {
142  log_warnf( 0, WARNING_SUSPICIOUS, "Unable to load dynamic library '%s': %s", name, dlerror() );
143  error_context_pop();
144  return 0;
145  }
146 
147 #else
148 
149  log_errorf( 0, ERROR_NOT_IMPLEMENTED, "Dynamic library loading not implemented for this platform: %s", name );
150  error_context_pop();
151  return 0;
152 
153 #endif
154 
155  id = objectmap_reserve( _library_map );
156  if( !id )
157  {
158 #if FOUNDATION_PLATFORM_WINDOWS
159  FreeLibrary( dll );
160 #elif FOUNDATION_PLATFORM_POSIX
161  dlclose( lib );
162 #endif
163  log_errorf( 0, ERROR_OUT_OF_MEMORY, "Unable to allocate new library '%s', map full", name );
164  error_context_pop();
165  return 0;
166  }
167  library = memory_allocate_zero( sizeof( library_t ), 0, MEMORY_PERSISTENT );
168  atomic_store32( &library->ref, 1 );
169  library->id = id;
170  library->namehash = string_hash( name );
171  string_copy( library->name, name, 32 );
172 #if FOUNDATION_PLATFORM_WINDOWS
173  library->dll = dll;
174 #elif FOUNDATION_PLATFORM_POSIX
175  library->lib = lib;
176 #endif
177  objectmap_set( _library_map, id, library );
178 
179  error_context_pop();
180 
181  return library->id;
182 }
