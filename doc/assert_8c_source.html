46 int assert_report( uint64_t context, const char* condition, const char* file, int line, const char* msg )
47 {
48  static const char nocondition[] = "<Static fail>";
49  static const char nofile[] = "<No file>";
50  static const char nomsg[] = "<No message>";
51  static const char assert_format[] = "****** ASSERT FAILED ******\nCondition: %s\nFile/line: %s : %d\n%s%s\n%s\n";
52 
53  if( !condition ) condition = nocondition;
54  if( !file ) file = nofile;
55  if( !msg ) msg = nomsg;
56 
57  if( _assert_handler && ( _assert_handler != assert_report ) )
58  return (*_assert_handler)( context, condition, file, line, msg );
59 
60  _assert_context_buffer[0] = 0;
61  error_context_buffer( _assert_context_buffer, ASSERT_BUFFER_SIZE );
62 
63  _assert_stacktrace_buffer[0] = 0;
64  if( foundation_is_initialized() )
65  {
66  if( stacktrace_capture( _assert_stacktrace, ASSERT_STACKTRACE_MAX_DEPTH, ASSERT_STACKTRACE_SKIP_FRAMES ) > 0 )
67  {
68  //TODO: Resolve directly into buffer to avoid memory allocations in assert handler
69  char* trace = stacktrace_resolve( _assert_stacktrace, ASSERT_STACKTRACE_MAX_DEPTH, 0U );
70  string_copy( _assert_stacktrace_buffer, trace, ASSERT_BUFFER_SIZE );
71  string_deallocate( trace );
72  }
73  }
74  else
75  {
76  string_copy( _assert_stacktrace_buffer, "<no stacktrace - not initialized>", ASSERT_BUFFER_SIZE );
77  }
78 
79  snprintf( _assert_box_buffer, (size_t)ASSERT_BUFFER_SIZE, assert_format, condition, file, line, _assert_context_buffer, msg, _assert_stacktrace_buffer );
80 
81  log_errorf( context, ERROR_ASSERT, "%s", _assert_box_buffer );
82 
83  system_message_box( "Assert Failure", _assert_box_buffer, false );
84 
85  return 1;
86 }
