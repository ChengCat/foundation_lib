182 static FOUNDATION_FORCEINLINE FOUNDATION_PURECALL void
183 byteorder_swap(void* buffer, const size_t size) {
184  size_t i, num;
185  char* src = (char*)buffer;
186  char* dst = src + (size - 1);
187  for (i = 0, num = size / 2; i < num; ++i, --dst, ++src) {
188  char tmp = *dst;
189  *dst = *src;
190  *src = tmp;
191  }
192 }
