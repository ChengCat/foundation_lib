99 static FORCEINLINE PURECALL void byteorder_swap( void* RESTRICT buffer, const int size )
100 {
101  int i, num;
102  char* RESTRICT src = buffer;
103  char* RESTRICT dst = src + ( size - 1 );
104  for( i = 0, num = size / 2; i < num; ++i, --dst, ++src )
105  {
106  char tmp = *dst;
107  *dst = *src;
108  *src = tmp;
109  }
110 }
