63 static void _crash_create_mini_dump( EXCEPTION_POINTERS* pointers, const char* name, char* dump_file )
64 {
65  MINIDUMP_EXCEPTION_INFORMATION info;
66 
67  HANDLE file;
68  SYSTEMTIME local_time;
69 
70  GetLocalTime( &local_time );
71 
72  dump_file[0] = 0;
73  string_format_buffer( dump_file, FOUNDATION_MAX_PATHLEN + 128, "%s/%s-%04d%02d%02d-%02d%02d%02d-%ld-%ld.dmp",
74  environment_temporary_directory(), name ? name : string_from_uuid_static( environment_application()->instance ),
75  local_time.wYear, local_time.wMonth, local_time.wDay,
76  local_time.wHour, local_time.wMinute, local_time.wSecond,
77  GetCurrentProcessId(), GetCurrentThreadId());
78  fs_make_directory( environment_temporary_directory() );
79  file = CreateFileA( dump_file, GENERIC_WRITE, FILE_SHARE_WRITE | FILE_SHARE_READ, 0, CREATE_ALWAYS, 0, 0 );
80 
81  if( file && ( file != INVALID_HANDLE_VALUE ) )
82  {
83  BOOL success = FALSE;
84  HMODULE lib = LoadLibraryA( "dbghelp.dll" );
85 
86  if( lib )
87  {
88  MiniDumpWriteDumpFn CallMiniDumpWriteDump = (MiniDumpWriteDumpFn)GetProcAddress( lib, "MiniDumpWriteDump" );
89  if( CallMiniDumpWriteDump )
90  {
91  info.ThreadId = GetCurrentThreadId();
92  info.ExceptionPointers = pointers;
93  info.ClientPointers = TRUE;
94 
95  success = CallMiniDumpWriteDump( GetCurrentProcess(), GetCurrentProcessId(), file, MiniDumpWithDataSegs | MiniDumpWithProcessThreadData | MiniDumpWithThreadInfo, &info, 0, 0 );
96  if( !success )
97  dump_file[0] = 0;
98  }
99 
100  FreeLibrary( lib );
101  }
102  if( success )
103  {
104  FlushFileBuffers( file );
105  }
106 
107  CloseHandle( file );
108  }
109 }

189 int crash_guard( crash_guard_fn fn, void* data, crash_dump_callback_fn callback, const char* name )
190 {
191 #if FOUNDATION_PLATFORM_WINDOWS
192 
193 # if FOUNDATION_COMPILER_MSVC || FOUNDATION_COMPILER_INTEL || FOUNDATION_COMPILER_CLANG
194  __try
195  {
196  return fn( data );
197  }
198  __except( _crash_create_mini_dump( GetExceptionInformation(), name, _crash_dump_file ), EXCEPTION_EXECUTE_HANDLER )
199  {
200  if( callback )
201  callback( _crash_dump_file );
202 #if BUILD_ENABLE_ERROR_CONTEXT
203  _error_context_clear();
204 #endif
205  return CRASH_DUMP_GENERATED;
206  }
207 # else
208  SetUnhandledExceptionFilter( _crash_exception_filter );
209  _crash_exception_closure.callback = callback;
210  _crash_exception_closure.name = name;
211  return fn( data );
212 # endif
213 
214 #elif FOUNDATION_PLATFORM_POSIX
215  sigjmp_buf guard_env = {0};
216 
217  struct sigaction action;
218  memset( &action, 0, sizeof( action ) );
219 
220  //Signals we process globally
221  action.sa_sigaction = _crash_guard_sigaction;
222  action.sa_flags = SA_SIGINFO;
223  if( ( sigaction( SIGILL, &action, 0 ) < 0 ) ||
224  ( sigaction( SIGFPE, &action, 0 ) < 0 ) ||
225  ( sigaction( SIGSEGV, &action, 0 ) < 0 ) ||
226  ( sigaction( SIGBUS, &action, 0 ) < 0 ) ||
227  ( sigaction( SIGSYS, &action, 0 ) < 0 ) )
228  {
229  log_warn( 0, WARNING_SYSTEM_CALL_FAIL, "Unable to set crash guard signal actions" );
230  return fn( data );
231  }
232 
233  set_thread_crash_callback( callback );
234  set_thread_crash_callback_name( name );
235 
236  int ret = sigsetjmp( guard_env, 1 );
237  if( ret == 0 )
238  {
239  set_thread_crash_env( guard_env );
240  return fn( data );
241  }
242  return ret;
243 
244 #else
245 
246  //No guard mechanism in place yet for this platform
247  return fn( data );
248 
249 #endif
250 }
