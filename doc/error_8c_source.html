59 void _error_context_push( const char* name, const char* data )
60 {
61  error_context_t* context = get_thread_error_context();
62  if( !context )
63  {
64  context = memory_allocate_zero( sizeof( error_context_t ), 0, MEMORY_PERSISTENT );
65  set_thread_error_context( context );
66  }
67  FOUNDATION_ASSERT_MSG( context->depth < BUILD_SIZE_ERROR_CONTEXT_DEPTH, "Error context stack overflow" );
68  context->frame[ context->depth ].name = name;
69  context->frame[ context->depth ].data = data;
70  if( context->depth < BUILD_SIZE_ERROR_CONTEXT_DEPTH-1 )
71  ++context->depth;
72 }

94 void _error_context_buffer( char* buffer, unsigned int size )
95 {
96  error_context_t* context = get_thread_error_context();
97  if( context )
98  {
99  int i, written;
100  error_frame_t* frame = context->frame;
101  for( i = 0; size && ( i < context->depth ); ++i, ++frame )
102  {
103  written = snprintf( buffer, size, "When %s: %s\n", frame->name ? frame->name : "<something>", frame->data ? frame->data : "" );
104  if( ( written > 0 ) && ( written <= (int)size ) )
105  {
106  buffer += written;
107  size -= written;
108  }
109  else
110  break;
111  }
112  }
113 }
