37 static void _event_post_delay_with_flag( event_stream_t* stream, uint16_t id, uint16_t size, uint64_t object, const void* payload, uint16_t flags, uint64_t timestamp )
38 {
39  event_block_t* block;
40  event_t* event;
41  bool restored_block;
42  uint32_t basesize;
43  uint32_t allocsize;
44  int32_t last_write;
45 
46  //Events must have non-zero id
47  FOUNDATION_ASSERT_MSG( id, "Events must have non-zero id" );
48  if( !id )
49  return;
50 
51  //Events must be aligned to an even 8 bytes
52  basesize = sizeof( event_t ) + size;
53  if( basesize % 8 )
54  basesize += 8 - ( basesize % 8 );
55 
56  //Delayed events have extra 8 bytes payload to hold timestamp
57  allocsize = basesize;
58  if( timestamp )
59  allocsize += 8;
60 
61  //Lock the event block by atomic swapping the write block index
62  last_write = atomic_load32( &stream->write );
63  while( ( last_write < 0 ) || !atomic_cas32( &stream->write, EVENT_BLOCK_POSTING, last_write ) )
64  {
65  thread_yield();
66  last_write = atomic_load32( &stream->write );
67  }
68 
69  //We now have exclusive access to the event block
70  block = stream->block + last_write;
71 
72  if( ( block->used + allocsize + 2 ) >= block->capacity )
73  {
74 #define BLOCK_CHUNK_SIZE ( 32 * 1024 )
75  uint32_t prev_capacity = block->capacity + 2ULL;
76  if( block->capacity < BLOCK_CHUNK_SIZE )
77  {
78  block->capacity <<= 1;
79  block->capacity += allocsize;
80  }
81  else
82  {
83  block->capacity += BLOCK_CHUNK_SIZE;
84  FOUNDATION_ASSERT_MSG( block->capacity < BUILD_SIZE_EVENT_BLOCK_LIMIT, "Event stream block size > 4Mb" );
85  error_report( ERRORLEVEL_ERROR, ERROR_OUT_OF_MEMORY );
86  }
87  if( block->capacity % 16 )
88  block->capacity += 16 - ( basesize % 16 );
89  block->events = block->events ? memory_reallocate( block->events, block->capacity + 2ULL, 16, prev_capacity ) : memory_allocate( block->capacity + 2ULL, 16, MEMORY_PERSISTENT );
90  }
91 
92  event = pointer_offset( block->events, block->used );
93 
94  event->id = id;
95  event->serial = (uint16_t)( atomic_exchange_and_add32( &_event_serial, 1 ) & 0xFFFF );
96  event->size = allocsize;
97  event->flags = 0;
98  event->object = object;
99 
100  if( size )
101  memcpy( event->payload, payload, size );
102 
103  if( timestamp )
104  {
105  event->flags |= EVENTFLAG_DELAY;
106  *(uint64_t*)pointer_offset( event, basesize ) = timestamp;
107  }
108 
109  //Terminate with null id on next event
110  block->used += allocsize;
111  ((event_t*)pointer_offset( block->events, block->used ))->id = 0;
112 
113  //Now unlock the event block
114  restored_block = atomic_cas32( &stream->write, last_write, EVENT_BLOCK_POSTING );
115  FOUNDATION_ASSERT( restored_block );
116 }

134 event_t* event_next( const event_block_t* block, event_t* event )
135 {
136  uint64_t curtime = 0;
137  uint64_t eventtime;
138 
139  do
140  {
141  //Grab first event if no previous event, or grab next event
142  event = ( event ? pointer_offset( event, event->size ) : ( block && block->used ? block->events : 0 ) );
143  if( !event || !event->id )
144  return 0; // End of event list
145 
146  if( !( event->flags & EVENTFLAG_DELAY ) )
147  return event;
148 
149  if( !curtime )
150  curtime = time_current();
151 
152  eventtime = *(uint64_t*)pointer_offset( event, event->size - 8 );
153  if( eventtime <= curtime )
154  return event;
155 
156  //Re-post to next block
157  _event_post_delay_with_flag( block->stream, event->id, event->size - ( sizeof( event_t ) + 8 ), event->object, event->payload, event->flags, eventtime );
158  } while( true );
159 
160  return 0;
161 }

198 event_block_t* event_stream_process( event_stream_t* stream )
199 {
200  event_block_t* block;
201  bool restored_block;
202  int32_t last_write, new_write;
203 
204  if( !stream )
205  return 0;
206 
207  //Lock the write event block by atomic swapping the write block index
208  last_write = atomic_load32( &stream->write );
209  while( ( last_write < 0 ) || !atomic_cas32( &stream->write, EVENT_BLOCK_SWAPPING, last_write ) )
210  {
211  thread_yield();
212  last_write = atomic_load32( &stream->write );
213  }
214 
215  //Reset used on last read (safe, since read can only happen on one thread)
216  stream->block[ stream->read ].used = 0;
217 
218  //Swap blocks
219  new_write = stream->read;
220  stream->read = last_write;
221 
222  block = stream->block + last_write;
223 
224  //Unlock write event block
225  restored_block = atomic_cas32( &stream->write, new_write, EVENT_BLOCK_SWAPPING );
226  FOUNDATION_ASSERT( restored_block );
227 
228  return block;
229 }
