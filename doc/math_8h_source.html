557 static FOUNDATION_FORCEINLINE FOUNDATION_CONSTCALL type math_inc_wrap_##suffix( const type val, const type min, const type max ); \
558 static FOUNDATION_FORCEINLINE FOUNDATION_CONSTCALL type math_inc_wrap_##suffix( const type val, const type min, const type max ) { \
559  const type increased = val + 1; \
560  const type max_diff = max - val; \
561  const type max_diff_nz = (type)( ( (signed_type)max_diff | -(signed_type)max_diff ) >> bit_mask ); \

600 static FOUNDATION_FORCEINLINE FOUNDATION_CONSTCALL unsigned int
601 math_align_poweroftwo(unsigned int x) {
602  FOUNDATION_ASSERT(x > 1);
603 
604 #if FOUNDATION_COMPILER_INTEL && ( FOUNDATION_ARCH_X86 || FOUNDATION_ARCH_X86_64 )
605  --x;
606  __asm__("bsrl %1,%0"
607  :"=r"(x)
608  :"rm"(x));
609  return (1 << (x + 1));
610 #else
611  --x;
612  x |= x >> 1;
613  x |= x >> 2;
614  x |= x >> 4;
615  x |= x >> 8;
616  x |= x >> 16;
617  return ++x;
618 #endif
619 }
