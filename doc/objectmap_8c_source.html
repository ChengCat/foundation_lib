16 objectmap_t* objectmap_allocate( unsigned int size )
17 {
18  uint64_t bits;
19  unsigned int ip;
20  uintptr_t next_indexshift;
21  objectmap_t* map;
22  void** slot;
23 
24  FOUNDATION_ASSERT_MSG( size > 2, "Invalid objectmap size" );
25  if( size <= 2 )
26  size = 2;
27  bits = math_round( math_log2( (real)size ) ); //Number of bits needed
28  FOUNDATION_ASSERT_MSGFORMAT( bits < 50, "Invalid objectmap size %d", size );
29 
30  //Top two bits unused for Lua compatibility
31  map = memory_allocate_zero( sizeof( objectmap_t ) + ( sizeof( void* ) * size ), 16, MEMORY_PERSISTENT );
32  map->size_bits = bits;
33  map->id_max = ((1ULL<<(62ULL-bits))-1);
34  map->size = size;
35  map->mask_index = ((1ULL<<bits)-1ULL);
36  map->mask_id = ( 0x3FFFFFFFFFFFFFFFULL & ~map->mask_index );
37  atomic_store64( &map->free, 0 );
38  atomic_store64( &map->id, 1 );
39 
40  slot = map->map;
41  for( ip = 0, next_indexshift = 3; ip < ( size - 1 ); ++ip, next_indexshift += 2, ++slot )
42  *slot = (void*)next_indexshift;
43  *slot = (void*)((uintptr_t)-1);
44 
45  return map;
46 }

52

89 object_t objectmap_reserve( objectmap_t* map )
90 {
91  uint64_t idx, next, id;
92 
93  FOUNDATION_ASSERT( map ); /*lint -esym(613,pool) */
94 
95  //Reserve spot in array
96  //TODO: Look into double-ended implementation with allocation from tail and free push to head
97  do
98  {
99  idx = atomic_load64( &map->free );
100  if( idx >= map->size )
101  {
102  log_error( 0, ERROR_OUT_OF_MEMORY, "Pool full, unable to reserve id" );
103  return 0;
104  }
105  next = ((uintptr_t)map->map[idx]) >> 1;
106  } while( !atomic_cas64( &map->free, next, idx ) );
107 
108  //Sanity check that slot isn't taken
109  FOUNDATION_ASSERT_MSG( (intptr_t)(map->map[idx]) & 1, "Map failed sanity check, slot taken after reserve" );
110  map->map[idx] = 0;
111 
112  //Allocate ID
113  id = 0;
114  do
115  {
116  id = atomic_incr64( &map->id ) & map->id_max; //Wrap-around handled by masking
117  } while( !id );
118 
119  //Make sure id stays within correct bits (if fails, check objectmap allocation and the mask setup there)
120  FOUNDATION_ASSERT( ( ( id << map->size_bits ) & map->mask_id ) == ( id << map->size_bits ) );
121 
122  return ( id << map->size_bits ) | idx; /*lint +esym(613,pool) */
123 }

126 void objectmap_free( objectmap_t* map, object_t id )
127 {
128  uint64_t idx, last;
129 
130  FOUNDATION_ASSERT( map ); /*lint -esym(613,pool) */
131 
132  idx = (intptr_t)( id & map->mask_index );
133  if( (uintptr_t)map->map[idx] & 1 )
134  return; //Already free
135 
136  do
137  {
138  last = atomic_load64( &map->free );
139  map->map[idx] = (void*)((uintptr_t)(last<<1)|1);
140  } while( !atomic_cas64( &map->free, idx, last ) ); /*lint +esym(613,pool) */
141 }
