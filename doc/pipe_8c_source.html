39 stream_t* pipe_allocate( void )
40 {
41  stream_pipe_t* pipestream = memory_allocate_zero_context( HASH_STREAM, sizeof( stream_pipe_t ), 0, MEMORY_PERSISTENT );
42  stream_t* stream = (stream_t*)pipestream;
43 
44  _stream_initialize( stream, system_byteorder() );
45 
46  pipestream->type = STREAMTYPE_PIPE;
47  pipestream->path = string_format( "pipe://0x" PRIfixPTR, pipestream );
48  pipestream->mode = STREAM_OUT | STREAM_IN | STREAM_BINARY;
49  pipestream->sequential = true;
50 
51 #if FOUNDATION_PLATFORM_WINDOWS
52  {
53  //Inheritable by default so process can use for stdstreams
54  SECURITY_ATTRIBUTES security_attribs = {0};
55  security_attribs.nLength = sizeof( SECURITY_ATTRIBUTES );
56  security_attribs.bInheritHandle = TRUE;
57  security_attribs.lpSecurityDescriptor = 0;
58 
59  if( !CreatePipe( &pipestream->handle_read, &pipestream->handle_write, &security_attribs, 0 ) )
60  log_warnf( 0, WARNING_SYSTEM_CALL_FAIL, "Unable to create unnamed pipe: %s", system_error_message( GetLastError() ) );
61  }
62 #elif FOUNDATION_PLATFORM_POSIX
63  int fds[2] = { 0, 0 };
64  if( pipe( fds ) < 0 )
65  log_warnf( 0, WARNING_SYSTEM_CALL_FAIL, "Unable to create unnamed pipe: %s", system_error_message( 0 ) );
66  else
67  {
68  pipestream->fd_read = fds[0];
69  pipestream->fd_write = fds[1];
70  }
71 #endif
72 
73  pipestream->vtable = &_pipe_stream_vtable;
74 
75  return stream;
76 }

176 static uint64_t _pipe_stream_read( stream_t* stream, void* dest, uint64_t num )
177 {
178  stream_pipe_t* pipestream = (stream_pipe_t*)stream;
179  FOUNDATION_ASSERT( stream->type == STREAMTYPE_PIPE );
180 #if FOUNDATION_PLATFORM_WINDOWS
181  if( pipestream->handle_read && ( ( pipestream->mode & STREAM_IN ) != 0 ) )
182  {
183  uint64_t total_read = 0;
184  do
185  {
186  unsigned long num_read = 0;
187  if( !ReadFile( pipestream->handle_read, pointer_offset( dest, total_read ), (unsigned int)( num - total_read ), &num_read, 0 ) )
188  break;
189  total_read += num_read;
190  } while( total_read < num );
191  return total_read;
192  }
193 #elif FOUNDATION_PLATFORM_POSIX
194  if( pipestream->fd_read && ( ( pipestream->mode & STREAM_IN ) != 0 ) )
195  {
196  uint64_t total_read = 0;
197  do
198  {
199  ssize_t num_read = read( pipestream->fd_read, pointer_offset( dest, total_read ), (size_t)( num - total_read ) );
200  if( num_read < 0 )
201  break;
202  total_read += num_read;
203  } while( total_read < num );
204  return total_read;
205  }
206 #endif
207 
208  return 0;
209 }
