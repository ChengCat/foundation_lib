33 stream_t* buffer_stream_allocate( void* buffer, unsigned int mode, uint64_t size, uint64_t capacity, bool adopt, bool grow )
34 {
35  stream_buffer_t* buffer_stream = memory_allocate_zero_context( HASH_STREAM, (uint64_t)sizeof( stream_buffer_t ), 0, MEMORY_PERSISTENT );
36  stream_t* stream = (stream_t*)buffer_stream;
37 
38  _stream_initialize( stream, system_byteorder() );
39 
40  FOUNDATION_ASSERT_MSG( adopt || !grow, "Cannot grow buffer streams that are not adopted" );
41 
42  if( !buffer )
43  {
44  size = 0;
45  capacity = 0;
46  }
47 
48  if( size > capacity )
49  size = capacity;
50 
51  buffer_stream->type = STREAMTYPE_MEMORY;
52  buffer_stream->path = string_format( "buffer://0x%" PRIfixPTR, stream );
53  buffer_stream->mode = mode & ( STREAM_OUT | STREAM_IN | STREAM_BINARY );
54  buffer_stream->buffer = buffer;
55  buffer_stream->size = size;
56  buffer_stream->capacity = capacity;
57  buffer_stream->own = adopt;
58  buffer_stream->grow = ( adopt && grow );
59 
60  if( mode & STREAM_TRUNCATE )
61  buffer_stream->size = 0;
62  if( mode & STREAM_ATEND )
63  buffer_stream->current = buffer_stream->size;
64 
65  buffer_stream->vtable = &_buffer_stream_vtable;
66 
67  return stream;
68 }

79 static uint64_t _buffer_stream_read( stream_t* stream, void* dest, uint64_t num )
80 {
81  stream_buffer_t* buffer_stream = (stream_buffer_t*)stream;
82 
83  uint64_t available = buffer_stream->size - buffer_stream->current;
84  uint64_t num_read = ( num < available ) ? num : available;
85 
86  if( num_read > 0 )
87  {
88  memcpy( dest, pointer_offset( buffer_stream->buffer, buffer_stream->current ), (size_t)num_read );
89  buffer_stream->current += num_read;
90  return num_read;
91  }

97 static uint64_t _buffer_stream_write( stream_t* stream, const void* source, uint64_t num )
98 {
99  stream_buffer_t* buffer_stream = (stream_buffer_t*)stream;
100 
101  uint64_t available = buffer_stream->size - buffer_stream->current;
102  uint64_t want = num;
103  uint64_t num_write;
104 
105  if( want > available )
106  {
107  if( buffer_stream->capacity >= ( buffer_stream->current + want ) )
108  {
109  available = want;
110  buffer_stream->size = buffer_stream->current + want;
111  }
112  else if( buffer_stream->grow )
113  {
114  available = want;
115  buffer_stream->size = buffer_stream->current + want;
116  buffer_stream->capacity = ( buffer_stream->size < 1024 ) ? 1024 : buffer_stream->size + 1024;
117  buffer_stream->buffer = memory_reallocate( buffer_stream->buffer, buffer_stream->capacity, 0, buffer_stream->current ); //tail segment from current to size will be overwritten
118  }
119  else
120  {
121  available = buffer_stream->capacity - buffer_stream->current;
122  buffer_stream->size = buffer_stream->capacity;
123  }
124  }
125 
126  num_write = ( want < available ) ? want : available;
127  if( num_write > 0 )
128  {
129  memcpy( pointer_offset( buffer_stream->buffer, buffer_stream->current ), source, (size_t)num_write );
130  buffer_stream->current += num_write;
131  return num_write;
132  }
133 
134  return 0;
135 }

178 static void _buffer_stream_seek( stream_t* stream, int64_t offset, stream_seek_mode_t direction )
179 {
180  stream_buffer_t* buffer_stream = (stream_buffer_t*)stream;
181  int64_t new_current = 0;
182  if( direction == STREAM_SEEK_CURRENT )
183  new_current = (int64_t)buffer_stream->current + offset;
184  else if( direction == STREAM_SEEK_BEGIN )
185  new_current = offset;
186  else if( direction == STREAM_SEEK_END )
187  new_current = (int64_t)buffer_stream->size + offset;
188 
189  if( new_current < 0 )
190  buffer_stream->current = 0;
191  else if( new_current > (int64_t)buffer_stream->size )
192  buffer_stream->current = buffer_stream->size;
193  else
194  buffer_stream->current = (uint64_t)new_current;
195 }
