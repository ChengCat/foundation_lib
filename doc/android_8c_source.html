248 void _android_enable_sensor( int sensor_type )
249 {
250  FOUNDATION_ASSERT( sensor_type > 0 && sensor_type < 16 );
251  if( _android_sensor_enabled[sensor_type] )
252  return;
253 
254  ASensorManager* sensor_manager = ASensorManager_getInstance();
255  if( sensor_manager )
256  {
257  const ASensor* sensor = ASensorManager_getDefaultSensor( sensor_manager, sensor_type );
258  if( sensor )
259  {
260  log_debugf( 0, "Initializing sensor of type %d", sensor_type );
261  if( _android_sensor_queue )
262  {
263  if( ASensorEventQueue_enableSensor( _android_sensor_queue, sensor ) < 0 )
264  log_warnf( 0, WARNING_SYSTEM_CALL_FAIL, "Unable to enable sensor of type %d", sensor_type );
265  else
266  {
267  int min_delay = ASensor_getMinDelay( sensor );
268  if( min_delay < 60000 )
269  min_delay = 60000;
270  if( ASensorEventQueue_setEventRate( _android_sensor_queue, sensor, min_delay ) < 0 )
271  log_warnf( 0, WARNING_SYSTEM_CALL_FAIL, "Unable to set event rate %d for sensor of type %d", min_delay, sensor_type );
272 
273  _android_sensor_enabled[sensor_type] = true;
274  }
275  }
276  }
277  }
278  else
279  {
280  log_warnf( 0, WARNING_UNSUPPORTED, "Unable to initialize sensors, no sensor manager" );
281  }
282 }

286 {

288  if( !_android_sensor_enabled[sensor_type] )

292  if( _android_sensor_queue && sensor_manager )
293  {
294  const ASensor* sensor = ASensorManager_getDefaultSensor( sensor_manager, sensor_type );
295  if( sensor )
296  {
297  ASensorEventQueue_disableSensor( _android_sensor_queue, sensor );
298  _android_sensor_enabled[sensor_type] = false;
299  }
300  }
