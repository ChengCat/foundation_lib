37 {
38  if( bitbuffer->buffer < bitbuffer->end )

147 uint32_t bitbuffer_read32( bitbuffer_t* bitbuffer, unsigned int bits )
148 {
149  unsigned int ret;
150  unsigned int curbits;
151 
152  if( !bits )
153  return 0;
154 
155  FOUNDATION_ASSERT( bits <= 32 );
156  if( bits > 32 )
157  bits = 32;
158 
159  if( bitbuffer->offset_read >= 32 )
160  _bitbuffer_get( bitbuffer );
161 
162  curbits = 32 - bitbuffer->offset_read;
163  if( bits < curbits )
164  curbits = bits;
165 
166  ret = ( curbits == 32 ) ? bitbuffer->pending_read : ( ( bitbuffer->pending_read >> bitbuffer->offset_read ) & ( ( 1U << curbits ) - 1 ) );
167 
168  bitbuffer->offset_read += curbits;
169  bitbuffer->count_read += curbits;
170 
171  if( curbits == bits )
172  return ret;
173 
174  FOUNDATION_ASSERT( bits && curbits );
175  FOUNDATION_ASSERT( bitbuffer->offset_read == 32 );
176 
177  _bitbuffer_get( bitbuffer );
178 
179  ret |= ( bitbuffer->pending_read & ( ( 1U << ( bits - curbits ) ) - 1 ) ) << curbits;
180 
181  bitbuffer->offset_read = ( bits - curbits );
182  bitbuffer->count_read += ( bits - curbits );
183 
184  return ret;
185 }

244 void bitbuffer_write32( bitbuffer_t* bitbuffer, uint32_t value, unsigned int bits )
245 {
246  unsigned int curbits;
247 
248  if( !bits )
249  return;
250 
251  FOUNDATION_ASSERT( bits <= 32 );
252  if( bits > 32 )
253  bits = 32;
254 
255  curbits = 32 - bitbuffer->offset_write;
256  if( bits < curbits )
257  curbits = bits;
258 
259  bitbuffer->pending_write |= ( ( curbits == 32 ) ? value : ( value & ( ( 1U << curbits ) - 1 ) ) ) << bitbuffer->offset_write;
260  bitbuffer->offset_write += curbits;
261 
262  bitbuffer->count_write += bits;
263 
264  if( bitbuffer->offset_write == 32 )
265  _bitbuffer_put( bitbuffer );
266 
267  if( curbits == bits )
268  return;
269 
270  FOUNDATION_ASSERT( bits && curbits );
271 
272  bitbuffer->pending_write = ( value >> curbits ) & ( ( 1U << ( bits - curbits ) ) - 1 );
273  bitbuffer->offset_write = bits - curbits;
274 }
