227 bool mutex_wait( mutex_t* mutex, unsigned int timeout )
228 {
229 #if FOUNDATION_PLATFORM_WINDOWS
230  DWORD ret;
231 #elif FOUNDATION_PLATFORM_POSIX
232  struct timeval now;
233  struct timespec then;
234 #endif
235  FOUNDATION_ASSERT( mutex );
236 #if FOUNDATION_PLATFORM_WINDOWS
237 
238 #if !BUILD_DEPLOY
239  profile_wait( mutex->name );
240 #endif
241 
242  atomic_incr32( &mutex->waiting );
243 
244  ret = WaitForSingleObject( mutex->event, ( timeout == 0 ) ? INFINITE : timeout );
245 
246  if( ret == WAIT_OBJECT_0 )
247  mutex_lock( mutex );
248 
249  if( atomic_decr32( &mutex->waiting ) == 0 )
250  ResetEvent( mutex->event );
251 
252  return ret == WAIT_OBJECT_0;
253 
254 #elif FOUNDATION_PLATFORM_POSIX
255 
256  mutex_lock( mutex );
257 
258  if( mutex->pending )
259  {
260  mutex->pending = false;
261  return true;
262  }
263 
264  --mutex->lockcount;
265 
266  bool was_signal = false;
267  if( !timeout )
268  {
269  int ret = pthread_cond_wait( &mutex->cond, &mutex->mutex );
270  if( ret == 0 )
271  {
272  was_signal = true;
273  }
274  else
275  {
276  log_warnf( 0, WARNING_SYSTEM_CALL_FAIL, "Unable to wait on mutex '%s': %s (%d)", mutex->name, system_error_message( ret ), ret );
277  }
278  }
279  else
280  {
281  int ret;
282  gettimeofday( &now, 0 );
283  then.tv_sec = now.tv_sec + ( timeout / 1000 );
284  then.tv_nsec = ( now.tv_usec * 1000 ) + (long)( timeout % 1000 ) * 1000000L;
285  while( then.tv_nsec > 999999999 )
286  {
287  ++then.tv_sec;
288  then.tv_nsec -= 1000000000L;
289  }
290  ret = pthread_cond_timedwait( &mutex->cond, &mutex->mutex, &then );
291  if( ret == 0 )
292  {
293  was_signal = true;
294  }
295  else if( ret != ETIMEDOUT )
296  {
297  log_warnf( 0, WARNING_SYSTEM_CALL_FAIL, "Unable to wait (timed) on mutex '%s': %s (%d)", mutex->name, system_error_message( ret ), ret );
298  }
299  }
300 
301  ++mutex->lockcount;
302  mutex->lockedthread = thread_id();
303 
304  if( was_signal )
305  mutex->pending = false;
306  else
307  mutex_unlock( mutex );
308 
309  return was_signal;
310 
311 #else
312 # error mutex_wait not implemented
313 #endif
314 }
