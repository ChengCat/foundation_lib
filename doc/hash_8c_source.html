58 // Block mix - combine the key bits with the hash bits and scramble everything
59 #define bmix64( h1, h2, k1, k2, c1, c2 ) \
60  k1 *= c1; \
61  k1 = _rotl64(k1,23); \
62  k1 *= c2; \
63  h1 ^= k1; \
64  h1 += h2; \
65  h2 = _rotl64(h2,41); \
66  k2 *= c2; \
67  k2 = _rotl64(k2,23); \
68  k2 *= c1; \
69  h2 ^= k2; \
70  h2 += h1; \
71  h1 = h1*3+0x52dce729; \
72  h2 = h2*3+0x38495ab5; \
73  c1 = c1*5+0x7b7d159c; \
74  c2 = c2*5+0x6bce6396;
75 
76 //----------
77 // Finalization mix - avalanches all bits to within 0.05% bias
78 static FORCEINLINE uint64_t fmix64( uint64_t k )
79 {
80  k ^= k >> 33;
81  k *= 0xff51afd7ed558ccd;
82  k ^= k >> 33;
83  k *= 0xc4ceb9fe1a85ec53;
84  k ^= k >> 33;
85 
86  return k;
87 }

90 hash_t hash( const void* key, const unsigned int len )
91 {
92  const unsigned int nblocks = len / 16;
93  const uint64_t* blocks;
94  unsigned int i;
95  const uint8_t* tail;
96  uint64_t k1;
97  uint64_t k2;
98 
99  uint64_t h1 = 0x9368e53c2f6af274ULL ^ HASH_SEED;
100  uint64_t h2 = 0x586dcd208f7cd3fdULL ^ HASH_SEED;
101 
102  /*lint -esym(438,c1,c2) Last value of c1 and c2 not used */
103  uint64_t c1 = 0x87c37b91114253d5ULL;
104  uint64_t c2 = 0x4cf5ad432745937fULL;
105 
106  //----------
107  // body
108 
109  blocks = (const uint64_t*)key; /*lint !e826 Ok, loop below will not access data outside scope*/
110 
111 #if FOUNDATION_ARCH_ARM || FOUNDATION_ARCH_ARM_64
112  if( (uintptr_t)key % 8 )
113  for( i = 0; i < nblocks; ++i )
114  {
115  k1 = getblock_nonaligned(key,i*2);
116  k2 = getblock_nonaligned(key,i*2+1);
117 
118  bmix64(h1,h2,k1,k2,c1,c2);
119  }
120  else
121 #endif
122  for( i = 0; i < nblocks; ++i )
123  {
124  k1 = getblock(blocks,i*2);
125  k2 = getblock(blocks,i*2+1);
126 
127  bmix64(h1,h2,k1,k2,c1,c2);
128  }
129 
130  //----------
131  // tail
132 
133  tail = pointer_offset_const( key, nblocks * 16 );
134 
135  k1 = 0;
136  k2 = 0;
137 
138  switch(len & 15) /*lint -save -e616 -e825 -e744 */
139  {
140  case 15: k2 ^= ((uint64_t)tail[14]) << 48;
141  case 14: k2 ^= ((uint64_t)tail[13]) << 40;
142  case 13: k2 ^= ((uint64_t)tail[12]) << 32;
143  case 12: k2 ^= ((uint64_t)tail[11]) << 24;
144  case 11: k2 ^= ((uint64_t)tail[10]) << 16;
145  case 10: k2 ^= ((uint64_t)tail[ 9]) << 8;
146  case 9: k2 ^= ((uint64_t)tail[ 8]);
147 
148  case 8: k1 ^= ((uint64_t)tail[ 7]) << 56;
149  case 7: k1 ^= ((uint64_t)tail[ 6]) << 48;
150  case 6: k1 ^= ((uint64_t)tail[ 5]) << 40;
151  case 5: k1 ^= ((uint64_t)tail[ 4]) << 32;
152  case 4: k1 ^= ((uint64_t)tail[ 3]) << 24;
153  case 3: k1 ^= ((uint64_t)tail[ 2]) << 16;
154  case 2: k1 ^= ((uint64_t)tail[ 1]) << 8;
155  case 1: k1 ^= ((uint64_t)tail[ 0]);
156  bmix64(h1,h2,k1,k2,c1,c2);
157  }; /*lint -restore */
158 
159  //----------
160  // finalization
161 
162  h2 ^= len;
163 
164  h1 += h2;
165  h2 += h1;
166 
167  h1 = fmix64(h1);
168  h2 = fmix64(h2);
169 
170  h1 += h2;
171  /* We only need 64-bit part
172  h2 += h1;
173 
174  ((uint64_t*)out)[0] = h1;
175  ((uint64_t*)out)[1] = h2;*/
176  return h1;
177 }

207 void _static_hash_store( const void* key, const unsigned int len, const hash_t value )
208 {
209  char* stored;
210 
211  if( !_hash_lookup )
212  return;
213 
214  stored = (char*)((uintptr_t)hashtable64_get( _hash_lookup, value ));
215  if( stored )
216  {
217  FOUNDATION_ASSERT_MSG( string_equal_substr( stored, key, len ), "Static hash collision" );
218  FOUNDATION_ASSERT_MSG( string_length( stored ) == len, "Static hash collision" );
219  return;
220  }

222  stored = memory_allocate_context( 0, len + 1, 0, MEMORY_PERSISTENT );
223  memcpy( stored, key, len );
224  stored[len] = 0;
