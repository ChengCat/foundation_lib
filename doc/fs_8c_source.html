70 void fs_monitor( const char* path )
71 {
72  int mi;
73  char* path_clone = 0;
74 
75  //TODO: Full thread safety
76 
77  for( mi = 0; mi < BUILD_SIZE_FS_MONITORS;++mi )
78  {
79  if( string_equal( atomic_loadptr( &_fs_monitors[mi].path ), path ) )
80  return;
81  }
82 
83  memory_context_push( HASH_STREAM );
84 
85  path_clone = path_clean( string_clone( path ), path_is_absolute( path ) );
86 
87  for( mi = 0; mi < BUILD_SIZE_FS_MONITORS; ++mi )
88  {
89  if( !_fs_monitors[mi].thread && !atomic_loadptr( &_fs_monitors[mi].path ) && !_fs_monitors[mi].signal )
90  {
91  if( atomic_cas_ptr( &_fs_monitors[mi].path, path_clone, 0 ) )
92  {
93  _fs_monitors[mi].thread = thread_create( _fs_monitor, "fs_monitor", THREAD_PRIORITY_BELOWNORMAL, 0 );
94  _fs_monitors[mi].signal = mutex_allocate( "fs_monitor_signal" );
95  thread_start( _fs_monitors[mi].thread, _fs_monitors + mi );
96  break;
97  }
98  }
99  }
100 
101  if( mi == BUILD_SIZE_FS_MONITORS )
102  log_errorf( 0, ERROR_OUT_OF_MEMORY, "Unable to monitor file system, no free monitor slots: %s", path );
103 
104  memory_context_pop();
105 }

195 char** fs_subdirs( const char* path )
196 {
197  char** arr = 0;
198 #if FOUNDATION_PLATFORM_WINDOWS
199 
200  //Windows specific implementation of directory listing
201  HANDLE find;
202  WIN32_FIND_DATAW data;
203  char* pattern = path_append( string_clone( path ), "*" );
204  wchar_t* wpattern = wstring_allocate_from_string( pattern, 0 );
205  string_deallocate( pattern );
206 
207  memory_context_push( HASH_STREAM );
208 
209  find = FindFirstFileW( wpattern, &data );
210  if( find != INVALID_HANDLE_VALUE ) do
211  {
212  if( data.dwFileAttributes & FILE_ATTRIBUTE_DIRECTORY )
213  {
214  if( data.cFileName[0] == L'.' )
215  {
216  if( !data.cFileName[1] || ( data.cFileName[1] == L'.' ) )
217  continue; //Don't include . and .. directories
218  }
219  array_push( arr, string_allocate_from_wstring( data.cFileName, 0 ) );
220  }
221  } while( FindNextFileW( find, &data ) );
222  FindClose( find );
223 
224  memory_context_pop();
225 
226  wstring_deallocate( wpattern );
227 
228 #elif FOUNDATION_PLATFORM_POSIX
229 
230  //POSIX specific implementation of directory listing
231  DIR* dir = opendir( path );
232  if( dir )
233  {
234  struct dirent* entry = 0;
235  struct stat st;
236 
237  memory_context_push( HASH_STREAM );
238 
239  while( ( entry = readdir( dir ) ) != 0 )
240  {
241  if( entry->d_name[0] == '.' )
242  {
243  if( !entry->d_name[1] || ( entry->d_name[1] == '.' ) )
244  continue; //Don't include . and .. directories
245  }
246  char* found = path_append( string_clone( path ), entry->d_name );
247  if( !stat( found, &st ) && S_ISDIR( st.st_mode ) )
248  array_push( arr, string_clone( entry->d_name ) );
249  string_deallocate( found );
250  }
251  closedir( dir );
252 
253  memory_context_pop();
254  }
255 
256 #else
257 # error Not implemented
258 #endif
259 
260  return arr;
261 }

342 bool fs_remove_directory( const char* path )
343 {
344  bool result = false;
345  char* fpath = path_make_absolute( path );
346  char** subpaths;
347  char** subfiles;
348 #if FOUNDATION_PLATFORM_WINDOWS
349  wchar_t* wfpath = 0;
350 #endif
351  int i, num;
352 
353  if( !fs_is_directory( fpath ) )
354  goto end;
355 
356  subpaths = fs_subdirs( fpath );
357  for( i = 0, num = array_size( subpaths ); i < num; ++i )
358  {
359  char* next = path_merge( fpath, subpaths[i] );
360  fs_remove_directory( next );
361  string_deallocate( next );
362  }
363  string_array_deallocate( subpaths );
364 
365  subfiles = fs_files( fpath );
366  for( i = 0, num = array_size( subfiles ); i < num; ++i )
367  {
368  char* next = path_merge( fpath, subfiles[i] );
369  fs_remove_file( next );
370  string_deallocate( next );
371  }
372  string_array_deallocate( subfiles );
373 
374 #if FOUNDATION_PLATFORM_WINDOWS
375  wfpath = wstring_allocate_from_string( fpath, 0 );
376  result = RemoveDirectoryW( wfpath );
377  wstring_deallocate( wfpath );
378 #elif FOUNDATION_PLATFORM_POSIX
379  result = ( rmdir( fpath ) == 0 );
380 #else
381 # error Not implemented
382 #endif
383 
384  end:
385 
386  string_deallocate( fpath );
387 
388  return result;
389 }

392 bool fs_make_directory( const char* path )
393 {
394  char* fpath;
395  char** paths;
396  char* curpath;
397  int ipath;
398  int pathsize;
399  bool result = true;
400 
401  fpath = path_make_absolute( path );
402  paths = string_explode( fpath, "/", false );
403  pathsize = array_size( paths );
404  curpath = 0;
405  ipath = 0;
406 
407  memory_context_push( HASH_STREAM );
408 
409 #if FOUNDATION_PLATFORM_WINDOWS
410  if( pathsize )
411  {
412  char* first = paths[ipath];
413  unsigned int flen = string_length( first );
414  if( flen && ( first[ flen - 1 ] == ':' ) )
415  {
416  curpath = string_clone( first );
417  ++ipath; //Drive letter
418  }
419  }
420 #endif
421 
422  for( ; ipath < pathsize; ++ipath )
423  {
424  curpath = string_append( curpath, "/" );
425  curpath = string_append( curpath, paths[ipath] );
426 
427  if( !fs_is_directory( curpath ) )
428  {
429 #if FOUNDATION_PLATFORM_WINDOWS
430  wchar_t* wpath = wstring_allocate_from_string( curpath, 0 );
431  result = CreateDirectoryW( wpath, 0 );
432  wstring_deallocate( wpath );
433 #elif FOUNDATION_PLATFORM_POSIX
434  mode_t mode = S_IRUSR | S_IWUSR | S_IXUSR | S_IRGRP | S_IWGRP | S_IXGRP | S_IROTH | S_IXOTH;
435  result = ( mkdir( curpath, mode ) == 0 );
436 #else
437 # error Not implemented
438 #endif
439  if( !result )
440  {
441  log_warnf( 0, WARNING_SUSPICIOUS, "Failed to create directory: %s", curpath );
442  //Unable to create directory
443  goto end;
444  }
445 
446  log_debugf( 0, "Created directory: %s", curpath );
447  }
448  }
449 
450  end:
451 
452  string_deallocate( fpath );
453  string_array_deallocate( paths );
454  string_deallocate( curpath );
455 
456  memory_context_pop();
457 
458  return result;
459 }

577 char** fs_matching_files( const char* path, const char* pattern, bool recurse )
578 {
579  char** names = 0;
580  char** subdirs = 0;
581  unsigned int id, dsize, in, nsize;
582 
583 #if FOUNDATION_PLATFORM_WINDOWS
584 
585  //Windows specific implementation of directory listing
586  WIN32_FIND_DATAW data;
587  char* pathpattern = path_merge( path, pattern );
588  wchar_t* wpattern = wstring_allocate_from_string( pathpattern, 0 );
589 
590  HANDLE find = FindFirstFileW( wpattern, &data );
591 
592  memory_context_push( HASH_STREAM );
593 
594  if( find != INVALID_HANDLE_VALUE ) do
595  {
596  if( !( data.dwFileAttributes & FILE_ATTRIBUTE_DIRECTORY ) )
597  array_push( names, path_clean( string_allocate_from_wstring( data.cFileName, 0 ), false ) );
598  } while( FindNextFileW( find, &data ) );
599 
600  memory_context_pop();
601 
602  FindClose( find );
603 
604  wstring_deallocate( wpattern );
605  string_deallocate( pathpattern );
606 
607 #else
608 
609  char** fnames = fs_files( path );
610 
611  memory_context_push( HASH_STREAM );
612 
613  for( in = 0, nsize = array_size( fnames ); in < nsize; ++in )
614  {
615  if( string_match_pattern( fnames[in], pattern ) )
616  {
617  array_push( names, fnames[in] );
618  fnames[in] = 0;
619  }
620  }
621 
622  memory_context_pop();
623 
624  string_array_deallocate( fnames );
625 
626 #endif
627 
628  if( !recurse )
629  return names;
630 
631  subdirs = fs_subdirs( path );
632 
633  memory_context_push( HASH_STREAM );
634 
635  for( id = 0, dsize = array_size( subdirs ); id < dsize; ++id )
636  {
637  char* subpath = path_merge( path, subdirs[id] );
638  char** subnames = fs_matching_files( subpath, pattern, true );
639 
640  for( in = 0, nsize = array_size( subnames ); in < nsize; ++in )
641  array_push( names, path_merge( subdirs[id], subnames[in] ) );
642 
643  string_array_deallocate( subnames );
644  string_deallocate( subpath );
645  }
646 
647  memory_context_pop();
648 
649  string_array_deallocate( subdirs );
650 
651  return names;
652 }

678 static void _add_notify_subdir( int notify_fd, const char* path, fs_watch_t** watch_arr, char*** path_arr )
679 {
680  char** subdirs = 0;
681  char* local_path = 0;
682  int fd = inotify_add_watch( notify_fd, path, IN_CREATE | IN_DELETE | IN_MODIFY | IN_MOVE );
683  if( fd < 0 )
684  {
685  log_warnf( 0, WARNING_SYSTEM_CALL_FAIL, "Failed watching subdir: %s (%d)", path, fd );
686  return;
687  }
688 
689  //log_debugf( 0, "Watching subdir: %s (%d)", path, fd );
690 
691  //Include terminating / in paths stored in path_arr/watch_arr
692  local_path = string_format( "%s/", path ? path : "" );
693  array_push( (*path_arr), local_path );
694 
695  fs_watch_t watch;
696  watch.fd = fd;
697  watch.path = local_path;
698  array_push( (*watch_arr), watch );
699 
700  //Recurse
701  subdirs = fs_subdirs( local_path );
702  for( int i = 0, size = array_size( subdirs ); i < size; ++i )
703  {
704  char* subpath = path_merge( local_path, subdirs[i] );
705  _add_notify_subdir( notify_fd, subpath, watch_arr, path_arr );
706  string_deallocate( subpath );
707  }
708  string_array_deallocate( subdirs );

1009 #else

1011  //Not implemented yet, just wait for signal to simulate thread
1012  if( mutex_wait( monitor->signal, 0 ) )
1013  mutex_unlock( monitor->signal );
1014 #endif
1015  }
1016 
1017  log_debugf( 0, "Stopped monitoring file system: %s", monitor->path );
1018 
1019 #if FOUNDATION_PLATFORM_WINDOWS
1020

1022

1024 
1025  if( handles[1] )
1026  CloseHandle( handles[1] );
1027 
1028  memory_deallocate( buffer );
1029 
1030 #elif FOUNDATION_PLATFORM_LINUX || FOUNDATION_PLATFORM_ANDROID

1034  array_deallocate( watch );
1035 
1036 #elif FOUNDATION_PLATFORM_APPLE

1038  _fs_event_stream_destroy( event_stream );

1040 #endif

1338 stream_t* fs_open_file( const char* path, unsigned int mode )
1339 {

1344  unsigned int in_mode = mode;
1345 
1346  if( !path )
1347  return 0;
1348 
1349  if( !mode )
1350  mode = STREAM_IN | STREAM_BINARY;
1351  if( !( mode & STREAM_IN ) && !( mode & STREAM_OUT ) )
1352  mode |= STREAM_IN;
1353 
1354  file = memory_allocate_zero_context( HASH_STREAM, sizeof( stream_file_t ), 0, MEMORY_PERSISTENT );
1355  stream = GET_STREAM( file );
1356  _stream_initialize( stream, BUILD_DEFAULT_STREAM_BYTEORDER );
1357 
1358  stream->type = STREAMTYPE_FILE;
1359 
1360  abspath = path_make_absolute( path );
1361  dotrunc = false;
1362  has_protocol = string_equal_substr( abspath, "file://", 7 );
1363 
1364  file->fd = _fs_file_fopen( _fs_path( abspath ), mode, &dotrunc );
1365 
1366  if( ( mode & STREAM_OUT ) && !file->fd && !( mode & STREAM_TRUNCATE ) )
1367  {
1368  string_deallocate( abspath );
1369  stream_deallocate( stream );
1370  return fs_open_file( path, in_mode | STREAM_TRUNCATE );
1371  }
1372
