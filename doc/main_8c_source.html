24 BOOL STDCALL _main_console_handler( DWORD control_type )
25 {
26  const char* control_name = "UNKNOWN";
27  bool post_terminate = false;
28  bool handled = true;
29 
30  switch( control_type )
31  {
32  case CTRL_C_EVENT: control_name = "CTRL_C"; post_terminate = true; break;
33  case CTRL_BREAK_EVENT: control_name = "CTRL_BREAK"; break;
34  case CTRL_CLOSE_EVENT: control_name = "CTRL_CLOSE"; post_terminate = true; break;
35  case CTRL_LOGOFF_EVENT: control_name = "CTRL_LOGOFF"; post_terminate = !config_bool( HASH_APPLICATION, HASH_DAEMON ); break;
36  case CTRL_SHUTDOWN_EVENT: control_name = "CTRL_SHUTDOWN"; post_terminate = true; break;
37  default: handled = false; break;
38  }
39  log_infof( 0, "Caught console control: %s (%d)", control_name, control_type );
40  if( post_terminate )
41  {
42  unsigned long level = 0, flags = 0;
43 
44  system_post_event( FOUNDATIONEVENT_TERMINATE );
45 
46  GetProcessShutdownParameters( &level, &flags );
47  SetProcessShutdownParameters( level, SHUTDOWN_NORETRY );
48 
49  thread_sleep( 1000 );
50  }
51  return handled;
52 }
