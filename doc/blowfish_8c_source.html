375 void blowfish_initialize( blowfish_t* blowfish, const void* key, const unsigned int length )
376 {
377  unsigned int capped_length = ( length <= MAXKEY ) ? length : MAXKEY;
378  const unsigned char* ckey = key;
379  uint32_t data, ldata, hdata;
380  unsigned int isub, ikey, iword, ibox, ientry;
381 
382  memcpy( blowfish->parray, _blowfish_parray_init, 4 * SUBKEYS );
383  memcpy( blowfish->sboxes, _blowfish_sboxes_init, 4 * SBOXES * SBOXENTRIES );
384 
385  for( isub = 0, ikey = 0; isub < SUBKEYS; ++isub )
386  {
387  data = 0;
388  for( iword = 0; iword < 4; ++iword, ++ikey )
389  data = ( data << 8 ) | ckey[ ikey % capped_length ];
390 
391  blowfish->parray[isub] ^= data;
392  }
393 
394  ldata = 0;
395  hdata = 0;
396 
397  for( isub = 0; isub < SUBKEYS; isub += 2 )
398  {
399  _blowfish_encrypt_words( blowfish, &ldata, &hdata );
400 
401  /*lint --e{661,679} */

403  blowfish->parray[isub+1] = hdata;
