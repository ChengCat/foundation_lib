53 static void _environment_set_executable_paths( const char* executable_path )
54 {
55  unsigned int last_path = string_rfind( executable_path, '/', STRING_NPOS );
56  if( last_path != STRING_NPOS )
57  {
58  if( !string_length( _environment_executable_dir ) )
59  string_copy( _environment_executable_dir, executable_path, last_path + 1 );
60  if( !string_length( _environment_executable_name ) )
61  {
62  string_copy( _environment_executable_name, executable_path + last_path + 1, FOUNDATION_MAX_PATHLEN );
63  }
64  }
65  else
66  {
67  if( !string_length( _environment_executable_dir ) )
68  _environment_executable_dir[0] = 0;
69  if( !string_length( _environment_executable_name ) )
70  string_copy( _environment_executable_name, executable_path, FOUNDATION_MAX_PATHLEN );
71  }
72 #if FOUNDATION_PLATFORM_WINDOWS
73  last_path = string_length( _environment_executable_name );
74  if( ( last_path > 4 ) && ( string_equal( _environment_executable_name + ( last_path - 4 ), ".exe" ) || string_equal( _environment_executable_name + ( last_path - 4 ), ".EXE" ) ) )
75  _environment_executable_name[ last_path - 4 ] = 0;
76 #endif
77  string_copy( _environment_executable_path, executable_path, FOUNDATION_MAX_PATHLEN );
78 }

381 const char* environment_temporary_directory( void )
382 {
383  if( _environment_temp_dir[0] )
384  return _environment_temp_dir;
385 #if FOUNDATION_PLATFORM_WINDOWS
386  {
387  char* path;
388  wchar_t* wpath = memory_allocate_zero( sizeof( wchar_t ) * FOUNDATION_MAX_PATHLEN, 0, MEMORY_TEMPORARY );
389  GetTempPathW( FOUNDATION_MAX_PATHLEN, wpath );
390  path = path_clean( string_allocate_from_wstring( wpath, 0 ), true );
391  string_copy( _environment_temp_dir, path, FOUNDATION_MAX_PATHLEN );
392  string_deallocate( path );
393  memory_deallocate( wpath );
394  }
395 #endif
396 #if FOUNDATION_PLATFORM_ANDROID
397  //Use application external data path, or if that fails, internal data path
398  struct android_app* app = android_app();
399  const char* test_path[] = { app && app->activity ? app->activity->externalDataPath : 0, app && app->activity ? app->activity->internalDataPath : 0 };
400  for( int itest = 0; itest < 2; ++itest )
401  {
402  if( test_path[itest] && string_length( test_path[itest] ) )
403  {
404  char* temp_path = path_prepend( string_format( ".tmp-%s", string_from_uuid_static( uuid_generate_random() ) ), test_path[itest] );
405  stream_t* temp_stream = fs_open_file( temp_path, STREAM_OUT | STREAM_BINARY );
406  if( temp_stream )
407  {
408  stream_deallocate( temp_stream );
409 
410  string_copy( _environment_temp_dir, test_path[itest], FOUNDATION_MAX_PATHLEN );
411  unsigned int len = string_length( _environment_temp_dir );
412  if( !len || ( _environment_temp_dir[ len - 1 ] != '/' ) )
413  {
414  _environment_temp_dir[ len++ ] = '/';
415  _environment_temp_dir[ len ] = 0;
416  }
417  string_copy( _environment_temp_dir + len, ".tmp", FOUNDATION_MAX_PATHLEN - len );
418 
419  //Clear it from old files
420  fs_remove_file( _environment_temp_dir );
421  fs_remove_directory( _environment_temp_dir );
422  fs_make_directory( _environment_temp_dir );
423  }
424  string_deallocate( temp_path );
425  }
426  }
427 #endif
428 #if FOUNDATION_PLATFORM_POSIX
429  if( !_environment_temp_dir[0] )
430  {
431  string_copy( _environment_temp_dir, P_tmpdir, FOUNDATION_MAX_PATHLEN );
432  unsigned int len = string_length( _environment_temp_dir );
433  if( ( len > 1 ) && ( _environment_temp_dir[ len - 1 ] == '/' ) )
434  _environment_temp_dir[ len - 1 ] = 0;
435  }
436 #endif
437 #if !FOUNDATION_PLATFORM_ANDROID
438  if( _environment_app.config_dir )
439  {
440  unsigned int curlen = string_length( _environment_temp_dir );
441  unsigned int cfglen = string_length( _environment_app.config_dir );
442  if( ( curlen + cfglen + 2 ) < FOUNDATION_MAX_PATHLEN )
443  {
444  _environment_temp_dir[curlen] = '/';
445  memcpy( _environment_temp_dir + curlen + 1, _environment_app.config_dir, cfglen + 1 );
446  }
447  }
448 #endif
449  return _environment_temp_dir;
450 }

453 const char* environment_variable( const char* var )
454 {
455 #if FOUNDATION_PLATFORM_WINDOWS
456  unsigned int required;
457  wchar_t* key = wstring_allocate_from_string( var, 0 );
458  wchar_t val[FOUNDATION_MAX_PATHLEN]; val[0] = 0;
459  if( ( required = GetEnvironmentVariableW( key, val, FOUNDATION_MAX_PATHLEN ) ) > FOUNDATION_MAX_PATHLEN )
460  {
461  wchar_t* val_local = memory_allocate( sizeof( wchar_t ) * ( required + 2 ), 0, MEMORY_TEMPORARY );
462  val_local[0] = 0;
463  GetEnvironmentVariableW( key, val_local, required + 1 );
464  if( _environment_var )
465  string_deallocate( _environment_var );
466  _environment_var = string_allocate_from_wstring( val_local, 0 );
467  memory_deallocate( val_local );
468  }
469  else
470  {
471  if( _environment_var )
472  string_deallocate( _environment_var );
473  _environment_var = string_allocate_from_wstring( val, 0 );
474  }
475  wstring_deallocate( key );
476  return _environment_var;
477 #elif FOUNDATION_PLATFORM_POSIX
478  return getenv( var );
479 #else
480 # error Not implemented
481 #endif
482 }
