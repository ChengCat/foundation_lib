28 void _array_growfn( void* arr, int increment, int factor, int itemsize )
29 {
30  void** parr = (void**)arr;
31  int prev_capacity = *parr ? _array_rawcapacity(*parr) : 0;
32  int capacity = *parr ? ( factor * prev_capacity + increment ) : increment;
33  int prev_used_size = itemsize * prev_capacity;
34  int storage_size = itemsize * capacity;
35  uint64_t header_size = 4ULL * _array_header_size;
36  uint64_t prev_used_buffer_size = (unsigned int)prev_used_size + header_size;
37  uint64_t buffer_size = (unsigned int)storage_size + header_size;
38  int* buffer = *parr ? memory_reallocate( _array_raw( *parr ), buffer_size, ARRAY_DEFAULT_ALIGN, prev_used_buffer_size ) : memory_allocate( buffer_size, ARRAY_DEFAULT_ALIGN, MEMORY_PERSISTENT );
39  FOUNDATION_ASSERT_MSG( buffer, "Failed to reallocate array storage" );
40  if( buffer )
41  {
42  buffer[0] = capacity;
43  if( !*parr )
44  {
45  buffer[1] = 0;
46  buffer[2] = ARRAY_WATERMARK;
47  }
48  *parr = buffer + _array_header_size;
49  }
50 }
