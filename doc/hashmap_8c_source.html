34 hashmap_t* hashmap_allocate( unsigned int buckets, unsigned int bucketsize )
35 {
36  hashmap_t* map;

38 
39  if( buckets < HASHMAP_MINBUCKETS )
40  buckets = HASHMAP_MINBUCKETS;
41  if( bucketsize < HASHMAP_MINBUCKETSIZE )
42  bucketsize = HASHMAP_MINBUCKETSIZE;

44  map = memory_allocate( sizeof( hashmap_t ) + sizeof( hashmap_bucket_t ) * buckets, 0, MEMORY_PERSISTENT );

46  map->num_buckets = buckets;

68 void* hashmap_insert( hashmap_t* map, hash_t key, void* value )
69 {
70  unsigned int ibucket = (unsigned int)( key % map->num_buckets );
71  hashmap_bucket_t bucket = map->bucket[ibucket];
72  unsigned int inode, nsize;
73  for( inode = 0, nsize = array_size( bucket ); inode < nsize; ++inode )
74  {
75  if( bucket[inode].key == key )
76  {
77  void* prev = bucket[inode].value;
78  bucket[inode].value = value;
79  return prev;
