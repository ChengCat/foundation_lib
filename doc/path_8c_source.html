18 char* path_clean( char* path, bool absolute )
19 {
20  //Since this function is used a lot we want to perform as much operations
21  //in place instead of splicing up into a string array and remerge
22  char* replace;
23  char* inpath;
24  char* next;
25  unsigned int inlength, length, remain, protocollen, up, last_up, prev_up, driveofs;
26 
27  if( !path )
28  return string_allocate( 0 );
29 
30  inpath = path;
31  inlength = string_length( path );
32  protocollen = string_find_string( path, "://", 0 );
33  if( ( protocollen != STRING_NPOS ) && ( protocollen > 1 ) )
34  {
35  absolute = true;
36  protocollen += 3; //Also skip the "://" separator
37  inlength -= protocollen;
38  path += protocollen;
39  }
40  else
41  {
42  protocollen = 0;
43  }
44  length = inlength;
45  driveofs = 0;
46 
47  replace = path;
48  while( ( replace = strchr( replace, '\\' ) ) != 0 )
49  *replace++ = '/';
50 
51  remain = length;
52  replace = path;
53  while( ( next = strstr( replace, "/./" ) ) != 0 )
54  {
55  remain -= (unsigned int)( next - replace ) + 2;
56  length -= 2;
57  memmove( next + 1, next + 3, remain ); //Include terminating zero to avoid looping when string ends in "/./"
58  replace = next;
59  }
60 
61  remain = length;
62  replace = path;
63  while( ( next = strstr( replace, "//" ) ) != 0 )
64  {
65  remain -= (unsigned int)( next - replace ) + 1;
66  --length;
67  memmove( next + 1, next + 2, remain ); //Include terminating zero to avoid looping when string ends in "//"
68  replace = next;
69  }
70 
71  path[length] = 0;
72 
73  if( string_equal( path, "." ) )
74  {
75  length = 0;
76  path[0] = 0;
77  }
78  else if( length > 1 )
79  {
80  if( ( path[ length - 2 ] == '/' ) && ( path[ length - 1 ] == '.' ) )
81  {
82  path[ length - 2 ] = 0;
83  length -= 2;
84  }
85  if( string_equal( path, "." ) )
86  {
87  length = 0;
88  path[0] = 0;
89  }
90  else if( string_equal( path, "./" ) )
91  {
92  length = 1;
93  path[0] = '/';
94  path[1] = 0;
95  }
96  else if( ( length > 1 ) && ( path[0] == '.' ) && ( path[1] == '/' ) )
97  {
98  --length;
99  memmove( path, path + 1, length );
100  path[length] = 0;
101  }
102  }
103 
104  if( absolute )
105  {
106  if( !length )
107  {
108  if( !inlength )
109  {
110  inlength = 2;
111  inpath = memory_reallocate( inpath, inlength + protocollen + 1, 0, protocollen + 1 );
112  path = inpath + protocollen;
113  }
114  path[0] = '/';
115  path[1] = 0;
116  ++length;
117  }
118  else if( ( length >= 2 ) && ( path[1] == ':' ) )
119  {
120  driveofs = 2;
121 
122  //Make sure first character is upper case
123  if( ( path[0] >= 'a' ) && ( path[0] <= 'z' ) )
124  path[0] = ( path[0] - (char)( (int)'a' - (int)'A' ) );
125 
126  if( length == 2 )
127  {
128  if( inlength <= 2 )
129  {
130  inpath = memory_reallocate( inpath, inlength + 2 + protocollen + 1, 0, inlength + protocollen + 1 );
131  inlength += 2;
132  path = inpath + protocollen;
133  }
134  path[2] = '/';
135  ++length;
136  }
137  else if( path[2] != '/' )
138  {
139  //splice in slash in weird-format paths (C:foo/bar/...)
140  if( inlength < ( length + 1 ) )
141  {
142  //Need more space
143  inpath = memory_reallocate( inpath, length + protocollen + 2, 0, inlength + protocollen + 1 );
144  inlength = length + 1;
145  path = inpath + protocollen;
146  }
147 
148  memmove( path + 3, path + 2, length + 1 - 2 );
149  path[2] = '/';
150  ++length;
151  }
152  }
153  else if( !protocollen && ( path[0] != '/' ) )
154  {
155  //make sure capacity is enough to hold additional character
156  if( inlength < ( length + 1 ) )
157  {
158  //Need more space
159  inpath = memory_reallocate( inpath, length + protocollen + 2, 0, inlength + protocollen + 1 );
160  inlength = length + 1;
161  path = inpath + protocollen;
162  }
163 
164  memmove( path + 1, path, length + 1 );
165  path[0] = '/';
166  ++length;
167  }
168  }
169  else //relative
170  {
171  if( length && ( path[0] == '/' ) )
172  {
173  memmove( path, path + 1, length - 1 );
174  --length;
175  }
176  }
177 
178  //Deal with .. references
179  last_up = driveofs;
180  while( ( up = string_find_string( path, "/../", last_up ) ) != STRING_NPOS )
181  {
182  if( up >= length )
183  break;
184  if( up == driveofs )
185  {
186  if( absolute )
187  {
188  memmove( path + driveofs + 1, path + driveofs + 4, length - ( driveofs + 3 ) );
189  length -= 3;
190  }
191  else
192  {
193  last_up = driveofs + 3;
194  }
195  continue;
196  }
197  prev_up = string_rfind( path, '/', up - 1 );
198  if( prev_up == STRING_NPOS )
199  {
200  if( absolute )
201  {
202  memmove( path, path + up + 3, length - up - 2 );
203  length -= ( up + 3 );
204  }
205  else
206  {
207  memmove( path, path + up + 4, length - up - 3 );
208  length -= ( up + 4 );
209  }
210  }
211  else if( prev_up >= last_up )
212  {
213  memmove( path + prev_up, path + up + 3, length - up - 2 );
214  length -= ( up - prev_up + 3 );
215  }
216  else
217  {
218  last_up = up + 1;
219  }
220  }
221 
222  if( length > 1 )
223  {
224  if( path[ length - 1 ] == '/' )
225  {
226  path[ length - 1 ] = 0;
227  --length;
228  }
229  }
230 
231  if( protocollen )
232  {
233  if( path[0] == '/' )
234  {
235  if( length == 1 )
236  length = 0;
237  else
238  {
239  memmove( path, path + 1, length );
240  --length;
241  }
242  }
243  length += protocollen;
244  path = inpath;
245  }
246 
247  path[length] = 0;
248 
249  return path;
250 }

253 char* path_base_file_name( const char* path )
254 {
255  unsigned int start, end;
256  if( !path )
257  return string_allocate( 0 );
258  start = string_find_last_of( path, "/\\", STRING_NPOS );
259  end = string_find( path, '.', ( start != STRING_NPOS ) ? start : 0 );
260  //For "dot" files, i.e files with names like "/path/to/.file", return the dot name ".file"
261  if( !end || ( end == ( start + 1 ) ) )
262  end = STRING_NPOS;
263  if( start != STRING_NPOS )
264  return string_substr( path, start + 1, ( end != STRING_NPOS ) ? ( end - start - 1 ) : STRING_NPOS );
265  return string_substr( path, 0, end );
266 }

427 char* path_make_absolute( const char* path )
428 {
429  unsigned int up, last, length, protocollen;
430  char* abspath = string_clone( path );
431  if( !path_is_absolute( abspath ) )
432  {
433  abspath = string_prepend( abspath, "/" );
434  abspath = string_prepend( abspath, environment_current_working_directory() );
435  abspath = path_clean( abspath, true );
436  }
437  else
438  {
439  abspath = path_clean( abspath, true );
440  }
441 
442  protocollen = string_find_string( abspath, "://", 0 );
443  if( protocollen != STRING_NPOS )
444  protocollen += 3; //Also skip the "://" separator
445  else
446  protocollen = 0;
447 
448  //Deal with .. references
449  while( ( up = string_find_string( abspath, "/../", 0 ) ) != STRING_NPOS )
450  {
451  char* subpath;
452  if( ( protocollen && ( up == ( protocollen - 1 ) ) ) || ( !protocollen && ( up == 0 ) ) )
453  {
454  //This moves mem so "prot://../path" ends up as "prot://path"
455  memmove( abspath + protocollen, abspath + 3 + protocollen, string_length( abspath ) + 1 - ( 3 + protocollen ) );
456  continue;
457  }
458  last = string_rfind( abspath, '/', up - 1 );
459  if( last == STRING_NPOS )
460  {
461  //Must be a path like C:/../something since other absolute paths
462  last = up;
463  }
464  subpath = string_substr( abspath, 0, last );
465  subpath = string_append( subpath, abspath + up + 3 ); // +3 will include the / of the later part of the path
466  string_deallocate( abspath );
467  abspath = subpath;
468  }
469 
470  length = string_length( abspath );
471  if( length >= 3 )
472  {
473  while( ( length >= 3 ) && ( abspath[length-3] == '/' ) && ( abspath[length-2] == '.' ) && ( abspath[length-1] == L'.' ) )
474  {
475  //Step up
476  if( length == 3 )
477  {
478  abspath[1] = 0;
479  length = 1;
480  }
481  else
482  {
483  length = string_rfind( abspath, '/', length - 4 );
484  abspath[length] = 0;
485  }
486  }
487  }
488 
489  return abspath;
490 }
