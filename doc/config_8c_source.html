55 static int64_t _config_string_to_int( const char* str )
56 {
57  unsigned int length = string_length( str );
58  unsigned int first_nonnumeric;
59  unsigned int dot_position;
60  if( length < 2 )
61  return string_to_int64( str );
62 
63  first_nonnumeric = string_find_first_not_of( str, "0123456789.", 0 );
64  if( ( first_nonnumeric == ( length - 1 ) ) && ( ( str[ first_nonnumeric ] == 'm' ) || ( str[ first_nonnumeric ] == 'M' ) ) )
65  {
66  dot_position = string_find( str, '.', 0 );
67  if( dot_position != STRING_NPOS )
68  {
69  if( string_find( str, '.', dot_position + 1 ) == STRING_NPOS )
70  return (int64_t)( string_to_real( str ) * ( REAL_C( 1024.0 ) * REAL_C( 1024.0 ) ) );
71  return string_to_int64( str ); //More than one dot
72  }
73  return string_to_int64( str ) * ( 1024LL * 1024LL );
74  }
75  if( ( first_nonnumeric == ( length - 1 ) ) && ( ( str[ first_nonnumeric ] == 'k' ) || ( str[ first_nonnumeric ] == 'K' ) ) )
76  {
77  dot_position = string_find( str, '.', 0 );
78  if( dot_position != STRING_NPOS )
79  {
80  if( string_find( str, '.', dot_position + 1 ) == STRING_NPOS )
81  return (int64_t)( string_to_real( str ) * REAL_C( 1024.0 ) );
82  return string_to_int64( str ); //More than one dot
83  }
84  return string_to_int64( str ) * 1024LL;
85  }
86 
87  return string_to_int64( str );
88 }

125 static NOINLINE const char* _expand_environment( hash_t key, char* var )
126 {
127  if( key == HASH_EXECUTABLE_NAME )
128  return environment_executable_name();
129  else if( key == HASH_EXECUTABLE_DIRECTORY )
130  return environment_executable_directory();
131  else if( key == HASH_EXECUTABLE_PATH )
132  return environment_executable_path();
133  else if( key == HASH_INITIAL_WORKING_DIRECTORY )
134  return environment_initial_working_directory();
135  else if( key == HASH_CURRENT_WORKING_DIRECTORY )
136  return environment_current_working_directory();
137  else if( key == HASH_HOME_DIRECTORY )
138  return environment_home_directory();
139  else if( key == HASH_TEMPORARY_DIRECTORY )
140  return environment_temporary_directory();
141  else if( string_equal_substr( var, "variable[", 9 ) ) //variable[varname] - Environment variable named "varname"
142  {
143  const char* value;
144  unsigned int end_pos = string_find( var, ']', 9 );
145  if( end_pos != STRING_NPOS )
146  var[end_pos] = 0;
147  value = environment_variable( var );
148  if( end_pos != STRING_NPOS )

151  }

156 static NOINLINE char* _expand_string( hash_t section_current, char* str )
157 {
158  char* expanded;
159  char* variable;
160  unsigned int var_pos, var_end_pos, variable_length, separator, var_offset;
161  hash_t section, key;
162 
163  expanded = str;
164  var_pos = string_find_string( expanded, "$(", 0 );
165 
166  while( var_pos != STRING_NPOS )
167  {
168  var_end_pos = string_find( expanded, ')', var_pos + 2 );
169  FOUNDATION_ASSERT_MSG( var_end_pos != STRING_NPOS, "Malformed config variable statement" );
170  variable = string_substr( expanded, var_pos, ( var_end_pos != STRING_NPOS ) ? ( 1 + var_end_pos - var_pos ) : STRING_NPOS );
171 
172  section = section_current;
173  key = 0;
174  variable_length = string_length( variable );
175  separator = string_find( variable, ':', 0 );
176  if( separator != STRING_NPOS )
177  {
178  if( separator != 2 )
179  section = hash( variable + 2, separator - 2 );
180  var_offset = separator + 1;
181  }
182  else
183  {
184  var_offset = 2;
185  }
186  key = hash( variable + var_offset, variable_length - ( var_offset + ( variable[ variable_length - 1 ] == ')' ? 1 : 0 ) ) );
187 
188  if( expanded == str )
189  expanded = string_clone( str );
190 
191  if( section != HASH_ENVIRONMENT )
192  expanded = string_replace( expanded, variable, config_string( section, key ), false );
193  else
194  expanded = string_replace( expanded, variable, _expand_environment( key, variable + var_offset ), false );
195  string_deallocate( variable );
196 
197  var_pos = string_find_string( expanded, "$(", 0 );
198  }
199 #if BUILD_ENABLE_DEBUG_CONFIG
200  if( str != expanded )
201  log_debugf( HASH_CONFIG, "Expanded config value \"%s\" to \"%s\"", str, expanded );
202 #endif
203 
204  return expanded;
205 }

208 static NOINLINE void _expand_string_val( hash_t section, config_key_t* key )
209 {
210  bool is_true;
211  FOUNDATION_ASSERT( key->sval );
212  if( key->expanded != key->sval )
213  string_deallocate( key->expanded );
214  key->expanded = _expand_string( section, key->sval );
215 
216  is_true = string_equal( key->expanded, "true" );
217  key->bval = ( string_equal( key->expanded, "false" ) || string_equal( key->expanded, "0" ) || !string_length( key->expanded ) ) ? false : true;
218  key->ival = is_true ? 1 : _config_string_to_int( key->expanded );
219  key->rval = is_true ? REAL_C(1.0) : _config_string_to_real( key->expanded );
220 }

406

413  }
414  }

416  continue;

569 const char* config_string( hash_t section, hash_t key )
570 {
571  config_key_t* key_val = config_key( section, key, false );
572  if( !key_val )
573  return "";
574  //Convert to string
575  /*lint --e{788} We use default for remaining enums */
576  switch( key_val->type )
577  {
578  case CONFIGVALUE_BOOL: return key_val->bval ? "true" : "false";
579  case CONFIGVALUE_INT: if( !key_val->sval ) key_val->sval = string_from_int( key_val->ival, 0, 0 ); return key_val->sval;
580  case CONFIGVALUE_REAL: if( !key_val->sval ) key_val->sval = string_from_real( key_val->rval, 4, 0, '0' ); return key_val->sval;
581  default: break;
582  }
583  //String value of some form
584  if( !key_val->sval )
585  return "";
586  if( key_val->type >= CONFIGVALUE_STRING_VAR )
587  {
588  _expand_string_val( section, key_val );
589  return key_val->expanded;
590  }
591  return key_val->sval;
592 }

620 {

700 void config_parse( stream_t* stream, hash_t filter_section, bool overwrite )
701 {
702  char* buffer;
703  hash_t section = 0;
704  hash_t key = 0;
705  unsigned int line = 0;
706 
707 #if BUILD_ENABLE_DEBUG_CONFIG
708  log_debugf( HASH_CONFIG, "Parsing config stream: %s", stream_path( stream ) );
709 #endif
710  buffer = memory_allocate_zero( 1024ULL, 0, MEMORY_TEMPORARY );
711  while( !stream_eos( stream ) )
712  {
713  ++line;
714  stream_read_line_buffer( stream, buffer, 1024, '\n' );
715  string_strip( buffer, " \t\n\r" );
716  if( !string_length( buffer ) || ( buffer[0] == ';' ) || ( buffer[0] == '#' ) )
717  continue;
718  if( buffer[0] == '[' )
719  {
720  //Section declaration
721  unsigned int endpos = string_rfind( buffer, ']', string_length( buffer ) - 1 );
722  if( ( endpos == STRING_NPOS ) || ( endpos < 1 ) )
723  {
724  log_warnf( HASH_CONFIG, WARNING_BAD_DATA, "Invalid section declaration on line %d in config stream '%s'", line, stream_path( stream ) );
725  continue;
726  }
727  buffer[endpos] = 0;
728  section = hash( buffer + 1, endpos - 1 );
729 #if BUILD_ENABLE_DEBUG_CONFIG
730  log_debugf( HASH_CONFIG, " config: section set to '%s' (0x%llx)", buffer + 1, section );
731 #endif
732  }
733  else if( !filter_section || ( filter_section == section ) )
734  {
735  //name=value declaration
736  char* name;
737  char* value;
738  unsigned int separator = string_find( buffer, '=', 0 );
739  if( separator == STRING_NPOS )
740  {
741  log_warnf( HASH_CONFIG, WARNING_BAD_DATA, "Invalid value declaration on line %d in config stream '%s', missing assignment operator '=': %s", line, stream_path( stream ), buffer );
742  continue;
743  }
744 
745  name = string_strip_substr( buffer, " \t", separator );
746  value = string_strip( buffer + separator + 1, " \t" );
747  if( !string_length( name ) )
748  {
749  log_warnf( HASH_CONFIG, WARNING_BAD_DATA, "Invalid value declaration on line %d in config stream '%s', empty name string", line, stream_path( stream ) );
750  continue;
751  }
752 
753  key = hash( name, string_length( name ) );
754 
755  if( overwrite || !config_key( section, key, false ) )
756  {
757 #if BUILD_ENABLE_DEBUG_CONFIG
758  log_debugf( HASH_CONFIG, " config: %s (0x%llx) = %s", name, key, value );
759 #endif
760 
761  if( !string_length( value ) )
762  config_set_string( section, key, "" );
763  else if( string_equal( value, "false" ) )
764  config_set_bool( section, key, false );
765  else if( string_equal( value, "true" ) )
766  config_set_bool( section, key, true );
767  else if( ( string_find( value, '.', 0 ) != STRING_NPOS ) && ( string_find_first_not_of( value, "0123456789.", 0 ) == STRING_NPOS ) && ( string_find( value, '.', string_find( value, '.', 0 ) + 1 ) == STRING_NPOS ) ) //Exactly one "."
768  config_set_real( section, key, string_to_real( value ) );
769  else if( string_find_first_not_of( value, "0123456789", 0 ) == STRING_NPOS )
770  config_set_int( section, key, string_to_int64( value ) );
771  else
772  config_set_string( section, key, value );
773  }
774  }
775  }
776  memory_deallocate( buffer );
777 }

780 void config_parse_commandline( const char* const* cmdline, unsigned int num )
781 {
782  //TODO: Implement, format --section:key=value
783  unsigned int arg;
784  for( arg = 0; arg < num; ++arg )
785  {
786  if( string_match_pattern( cmdline[arg], "--*:*=*" ) )
787  {
788  unsigned int first_sep = string_find( cmdline[arg], ':', 0 );
789  unsigned int second_sep = string_find( cmdline[arg], '=', 0 );
790  if( ( first_sep != STRING_NPOS ) && ( second_sep != STRING_NPOS ) && ( first_sep < second_sep ) )
791  {
792  unsigned int section_length = first_sep - 2;
793  unsigned int end_pos = first_sep + 1;
794  unsigned int key_length = second_sep - end_pos;
795 
796  const char* section_str = cmdline[arg] + 2;
797  const char* key_str = pointer_offset_const( cmdline[arg], end_pos );
798 
799  hash_t section = hash( section_str, section_length );
800  hash_t key = hash( key_str, key_length );

802  char* value = string_substr( cmdline[arg], second_sep + 1, STRING_NPOS );
803  char* set_value = value;
804 
805  unsigned int value_length = string_length( value );
806 
807  if( !value_length )
808  config_set_string( section, key, "" );
809  else if( string_equal( value, "false" ) )
810  config_set_bool( section, key, false );
811  else if( string_equal( value, "true" ) )
812  config_set_bool( section, key, true );

817  else
