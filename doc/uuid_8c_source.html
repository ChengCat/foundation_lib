73 uuid_t uuid_generate_time( void )
74 {
75  uuid_time_t time_uuid;
76  uuid_convert_t convert;
77  int64_t current_time;
78  int32_t current_counter;
79  tick_t current_tick;
80  int in = 0;
81  uint32_t clock_seq = 0;
82  uint64_t host_id = 0;
83 
84  //Allows creation of 10000 unique timestamps per millisecond
85  current_time = time_system();
86  current_counter = atomic_incr32( &_uuid_last_counter ) % 10000;
87 
88  current_tick = ( (tick_t)current_time * 10000ULL ) + current_counter + 0x01B21DD213814000ULL; //Convert to 100ns since UUID UTC base time, October 15 1582, and add counter
89 
90  //We have no state so clock sequence is random
91  clock_seq = random32();
92 
93  time_uuid.time_low = (uint32_t)( current_tick & 0xFFFFFFFFULL );
94  time_uuid.time_mid = (uint16_t)( ( current_tick >> 32ULL ) & 0xFFFFULL );
95  time_uuid.time_hi_and_version = (uint16_t)( current_tick >> 48ULL );
96  time_uuid.clock_seq_low = ( clock_seq & 0xFF );
97  time_uuid.clock_seq_hi_and_reserved = ( ( clock_seq & 0x3F00 ) >> 8 );
98 
99  //If hardware node ID is null, use random and set identifier (multicast) bit
100  host_id = system_hostid();
101  if( host_id )
102  {
103  for( in = 0; in < 6; ++in )
104  time_uuid.node[5-in] = (uint8_t)( ( host_id >> ( 8ULL * in ) ) & 0xFF );
105  }
106  else
107  {
108  for( in = 0; in < 6; ++in )
109  time_uuid.node[in] = (uint8_t)( random32() & 0xFF );
110  time_uuid.node[0] |= 0x01;
111  }
112 
113  //Add variant and version
114  time_uuid.time_hi_and_version &= 0x0FFF;
115  time_uuid.time_hi_and_version |= ( 1 << 12 );
116  time_uuid.clock_seq_hi_and_reserved &= 0x3F;
117  time_uuid.clock_seq_hi_and_reserved |= 0x80;
118 
119  convert.time = time_uuid;
120  return convert.uuid;
121 }

124 uuid_t uuid_generate_name( const uuid_t namespace, const char* name )
125 {
126  //v3 uuid, namespace and md5
127  md5_t* md5;
128  uuid_raw_t namespace_id;
129  uuid_raw_t gen_uuid;
130  uuid_convert_t convert;
131  uint128_t digest;
132 
133  //Namespace in network byte order
134  convert.uuid = namespace;
135  namespace_id = convert.raw;
136  namespace_id.data1 = byteorder_bigendian32( namespace_id.data1 );
137  namespace_id.data2 = byteorder_bigendian16( namespace_id.data2 );
138  namespace_id.data3 = byteorder_bigendian16( namespace_id.data3 );
139 
140  md5 = md5_allocate();
141  md5_initialize( md5 );
142  md5_digest_raw( md5, &namespace_id, sizeof( namespace_id ) );
143  md5_digest( md5, name );
144  md5_finalize( md5 );
145 
146  //Convert to host order
147  digest = md5_get_digest_raw( md5 );
148  memcpy( &gen_uuid, &digest, sizeof( uuid_raw_t ) );
149  gen_uuid.data1 = byteorder_bigendian32( gen_uuid.data1 );
150  gen_uuid.data2 = byteorder_bigendian16( gen_uuid.data2 );
151  gen_uuid.data3 = byteorder_bigendian16( gen_uuid.data3 );
152 
153  //Add variant and version
154  gen_uuid.data3 &= 0x0FFF;
155  gen_uuid.data3 |= ( 3 << 12 ); //Variant 3 for MD5
156  gen_uuid.data4[0] &= 0x3F;
157  gen_uuid.data4[0] |= 0x80;
158 
159  md5_deallocate( md5 );
160 
161  convert.raw = gen_uuid;
162  return convert.uuid;
163 }
