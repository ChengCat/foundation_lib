314 static thread_return_t FOUNDATION_THREADCALL _thread_entry( thread_arg_t data )
315 {
316  uint64_t thr_osid;
317  uint64_t thr_id;
318  thread_t* thread = GET_THREAD_PTR( data );
319 
320  atomic_incr32( &thread->ref );
321  atomic_cas32( &thread->started, 1, 0 );
322  if( !atomic_cas32( &thread->running, 1, 0 ) )
323  {
324  log_warnf( 0, WARNING_SUSPICIOUS, "Unable to enter thread %llx, already running", thread->id );
325  _thread_dec_ref( thread );
326  return 0;
327  }
328 
329 #if FOUNDATION_PLATFORM_WINDOWS
330  thread->osid = GetCurrentThreadId();
331 #elif FOUNDATION_PLATFORM_LINUX || FOUNDATION_PLATFORM_ANDROID
332  pthread_t curid = pthread_self();
333  thread->osid = curid;
334 #if !BUILD_DEPLOY
335  prctl( PR_SET_NAME, thread->name, 0, 0, 0 );
336 #endif
337 #elif FOUNDATION_PLATFORM_MACOSX || FOUNDATION_PLATFORM_IOS
338  mach_port_t curid = pthread_mach_thread_np( pthread_self() );
339  thread->osid = curid;
340 #else
341  #error Not implemented
342 #endif
343  atomic_store32( &thread->terminate, 0 );
344 
345  set_thread_self( thread );
346 
347  FOUNDATION_ASSERT( atomic_load32( &thread->running ) == 1 );
348 
349  log_debugf( 0, "Started thread '%s' (%llx) ID %llx%s", thread->name, thread->osid, thread->id, crash_guard_callback() ? " (guarded)" : "" );
350 
351  if( system_debugger_attached() )
352  {
353  thread->result = thread->fn( thread->id, thread->arg );
354  }
355  else
356  {
357  int crash_result = crash_guard( _thread_guard_wrapper, thread, crash_guard_callback(), crash_guard_name() );
358  if( crash_result == CRASH_DUMP_GENERATED )
359  {
360  thread->result = (void*)((uintptr_t)CRASH_DUMP_GENERATED);
361  log_warnf( 0, WARNING_SUSPICIOUS, "Thread '%s' (%llx) ID %llx crashed", thread->name, thread->osid, thread->id );
362  }
363  }
364 
365  thr_osid = thread->osid;
366  thr_id = thread->id;
367  log_debugf( 0, "Terminated thread '%s' (%llx) ID %llx with %d refs", thread->name, thr_osid, thr_id, thread->ref );
368 
369  thread->osid = 0;
370 
371  set_thread_self( 0 );
372  thread_cleanup();
373 
374  if( !atomic_cas32( &thread->running, 0, 1 ) )
375  {
376  FOUNDATION_ASSERT_FAIL( "Unable to reset running flag" );
377  atomic_store32( &thread->running, 0 );
378  }
379 
380  log_debugf( 0, "Exiting thread '%s' (%llx) ID %llx with %d refs", thread->name, thr_osid, thr_id, thread->ref );
381 
382  _thread_dec_ref( thread );
383 
384  return 0;
385 }

388 bool thread_start( object_t id, void* data )
389 {
390 #if FOUNDATION_PLATFORM_WINDOWS
391  unsigned long osid = 0;
392 #endif
393  thread_t* thread = GET_THREAD( id );
394  if( !thread )
395  {
396  log_errorf( 0, ERROR_INVALID_VALUE, "Unable to start thread %llx, invalid id", id );
397  return false; //Old/invalid id
398  }
399 
400  if( atomic_load32( &thread->running ) > 0 )
401  {
402  log_warnf( 0, WARNING_SUSPICIOUS, "Unable to start thread %llx, already running", id );
403  return false; //Thread already running
404  }
405 
406  atomic_cas32( &thread->started, 0, 1 );
407 
408  thread->arg = data;
409 
410  if( !thread->stacksize )
411  thread->stacksize = BUILD_SIZE_DEFAULT_THREAD_STACK;
412 
413 #if FOUNDATION_PLATFORM_WINDOWS
414  thread->handle = CreateThread( 0, thread->stacksize, _thread_entry, thread, 0, &osid );
415  if( !thread->handle )
416  {
417  log_errorf( 0, ERROR_OUT_OF_MEMORY, "Unable to create thread: CreateThread failed: %s", system_error_message( GetLastError() ) );
418  return false;
419  }
420 #if !BUILD_DEPLOY
421  _set_thread_name( osid, thread->name );
422 #endif
423 #elif FOUNDATION_PLATFORM_POSIX
424  int err = pthread_create( &thread->thread, 0, _thread_entry, thread );
425  if( err )
426  {
427  log_errorf( 0, ERROR_OUT_OF_MEMORY, "Unable to create thread: pthread_create failed: %s", system_error_message( err ) );
428  return false;
429  }
430 #else
431 # error Not implemented
432 #endif
433 
434  return true;
435 }
