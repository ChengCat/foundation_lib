36 int _time_initialize( void )
37 {
38 #if FOUNDATION_PLATFORM_WINDOWS
39  tick_t unused;
40  if( !QueryPerformanceFrequency( (LARGE_INTEGER*)&_time_freq ) ||
41  !QueryPerformanceCounter( (LARGE_INTEGER*)&unused ) )
42  return -1;
43 #elif FOUNDATION_PLATFORM_APPLE
44  if( mach_timebase_info( &_time_info ) )
45  return -1;
46  _time_freq = 1000000000ULL;
47 #elif FOUNDATION_PLATFORM_POSIX
48  struct timespec ts = { .tv_sec = 0, .tv_nsec = 0 };
49  if( clock_gettime( CLOCK_MONOTONIC, &ts ) )
50  return -1;
51  _time_freq = 1000000000ULL;
52 #endif
53 
54  _time_oofreq = REAL_C(1.0) / (double)_time_freq;
55  _time_startup = time_current();
56 
57  return 0;
58 }
