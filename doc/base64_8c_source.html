76 unsigned int base64_decode( const char* src, void* dst, unsigned int srcsize, unsigned int dstsize )
77 {
78  unsigned int i, length, blocksize;
79  char* cdst = (char*)dst;
80  char* cdstend = ( dstsize ? cdst + dstsize : 0 );
81  length = srcsize ? srcsize : (unsigned int)string_length( src );
82  while( length )
83  {
84  unsigned char in[4] = { 0, 0, 0, 0 }; //Always build blocks of 4 bytes to decode, pad with 0
85  blocksize = 0;
86  for( i = 0; length && ( i < 4 ); i++ )
87  {
88  char v = 0;
89  while( length && !v ) //Consume one valid byte from input, discarding invalid data
90  {
91  v = *src++;
92  v = ( ( v < 43 || v > 122 ) ? 0 : _base64_decode[ v - 43 ] );
93  if( v )
94  {
95  in[i] = (unsigned char)( v - 62 );
96  blocksize++;
97  }
98  --length;
99  }
100  }
101  if( blocksize )
102  {
103  char out[3];
104  _decodeblock_base64( in, out );
105  for( i = 0; ( i < blocksize - 1 ) && ( !cdstend || ( cdst < cdstend ) ); ++i )
106  *cdst++ = out[i];
107  }
108  }
109
